group8                = "Enhancement 8: Daily Range Target Scaling"

// ===== SAFETY NET SETTINGS =====
groupSafety          = "Safety Net: Redundant Close"
safetyBars           = input.int(3,    "Safety Net: Bars to Resend Close", minval=1, maxval=10, group=groupSafety)

// ===== MACD/RSI (TUNED FOR 10-MIN EUSTX50) =====
macdFast   = input.int(8,    title="MACD Fast Length")
macdSlow   = input.int(17,   title="MACD Slow Length")
macdSignal = input.int(9,    title="MACD Signal Length")
rsiLength  = input.int(14,   title="RSI Length")
rsiBullMin = input.float(45, title="RSI Min for Long")
rsiBullMax = input.float(75, title="RSI Max for Long")

rsiBearMin = input.float(25, title="RSI Min for Short")


// ===== INDICATORS =====
ema200     = ta.ema(close, 200)
macdLine   = ta.ema(close, macdFast) - ta.ema(close, macdSlow)
signalLine = ta.ema(macdLine, macdSignal)
rsi        = ta.rsi(close, rsiLength)
atr        = ta.atr(14)
atrMA      = ta.sma(atr, 14)


// ===== MARKET DETECTION (FIX #8: was checking EUSTX50 twice) =====
isEustx = str.contains(syminfo.ticker, "EUSTX50") or str.contains(syminfo.ticker, "STOXX") or str.contains(syminfo.ticker, "SX5E") or str.contains(syminfo.ticker, "EU50")
isDax   = str.contains(syminfo.ticker, "GER40") or str.contains(syminfo.ticker, "DAX") or str.contains(syminfo.ticker, "DE40") or str.contains(syminfo.ticker, "GER")
isSpx   = str.contains(syminfo.ticker, "SPX") or str.contains(syminfo.ticker, "ES") or str.contains(syminfo.ticker, "SP500")


// ===== POSITION SIZING CALCULATION =====
float currentEquity  = strategy.equity
float riskAmount     = currentEquity * (riskPerTradePct / 100)

float stopPointsDummy = atr * hardStopAtr
float rawLots        = stopPointsDummy > 0 ? riskAmount / (stopPointsDummy * pointValue) : minLotSize
float calculatedLots = math.max(minLotSize, math.min(maxLotSize, math.floor(rawLots * 10) / 10))
volRatio             = atr / atrMA
volMultiplier        = math.min(volMultiplierMax, math.max(volMultiplierMin, 1.0 / volRatio))
float baseLots       = useDynamicSizing ? math.max(minLotSize, math.min(maxLotSize, math.floor(calculatedLots * volMultiplier * 10) / 10)) : calculatedLots
float scaleInLots    = math.max(minLotSize, math.floor(baseLots * (scaleInSizePercent / 100) * 10) / 10)


// ===== SESSION =====
currentHour  = hour(time, "Etc/UTC")
isEuroOpen   = (currentHour >= 7 and currentHour < 9)
isUSOpen     = (currentHour >= 13 and currentHour < 15)
isLondonOpen = (currentHour >= londonHourStart and currentHour < londonHourStart + 2)
isNYOpen     = (currentHour >= nyHourStart and currentHour < nyHourStart + 2)
isLunchHour  = (currentHour >= 11 and currentHour < 13)
isGoodHour   = volatileHours ? (isLondonOpen or isNYOpen) : true
lunchHourOk  = lunchAvoidance ? not isLunchHour : true
hourlyVolOk  = not useHourlyVol or (isGoodHour and lunchHourOk)


var float sessionMult = 1.0
if useSessionVolAdj
    if isEuroOpen
        sessionMult := euroOpenBoost
    else if isUSOpen
        sessionMult := usOpenBoost
    else
        sessionMult := 1.0
else
    sessionMult := 1.0


// ===== FILTERS =====
volatilityOk  = not useVolatilityFilter or (atr > 0.9 * atrMA)
htfEMA        = request.security(syminfo.tickerid, htfTF, ta.ema(close, 200))
htfBull       = close > htfEMA
htfBear       = close < htfEMA
recentHigh    = ta.highest(high, 20)
recentLow     = ta.lowest(low,  20)
sweptHigh     = high > recentHigh
sweptLow      = low  < recentLow
liquidityOk   = not useLiquiditySweep or sweptHigh or sweptLow
inSession     = not useSessionFilter or (currentHour >= 6 and currentHour <= 22)
prevClose     = request.security(syminfo.tickerid, "1D", close[1])
dayOpen       = request.security(syminfo.tickerid, "1D", open)
gapSize       = math.abs(dayOpen - prevClose)
isGapDay      = gapSize > (atr * gapThreshold)
barsSinceOpen = ta.barssince(timeframe.change("1D"))
gapFilterOk   = not useGapFilter or not isGapDay or barsSinceOpen > gapCooldownBars


var float effectiveRsiBullMinVal = rsiBullMin
var float effectiveRsiBullMaxVal = rsiBullMax
var float effectiveRsiBearMinVal = rsiBearMin
var float effectiveRsiBearMaxVal = rsiBearMax


if useIndexSpecificRSI
    if isDax
        effectiveRsiBullMinVal := daxRsiBullMin
        effectiveRsiBullMaxVal := daxRsiBullMax
        effectiveRsiBearMinVal := daxRsiBearMin
        effectiveRsiBearMaxVal := daxRsiBearMax
    else if isSpx
        effectiveRsiBullMinVal := spxRsiBullMin
        effectiveRsiBullMaxVal := spxRsiBullMax
        effectiveRsiBearMinVal := spxRsiBearMin
        effectiveRsiBearMaxVal := spxRsiBearMax


// ===== ECON EVENTS =====
var int[] eventTimes = array.new_int(0)
if barstate.isfirst
    array.push(eventTimes, int(economicEvents))
    array.push(eventTimes, int(economicEvents2))
    array.push(eventTimes, int(economicEvents3))
    array.push(eventTimes, int(economicEvents4))


var bool isEventSoon = false
isEventSoon := false
if avoidHighImpactEvents
    int eventCount = array.size(eventTimes)
    for i = 0 to eventCount - 1
        eventTime      = array.get(eventTimes, i)
        barsSinceEvent = (time - eventTime) / (timeframe.in_seconds() * 1000)
        barsUntilEvent = (eventTime - time) / (timeframe.in_seconds() * 1000)
        if (barsUntilEvent > 0 and barsUntilEvent <= cautionBefore) or (barsSinceEvent > 0 and barsSinceEvent <= cautionAfter)
            isEventSoon := true


// ===== VOLUME TREND =====
var float obvVal = 0.0
price_change = ta.change(close)
if price_change > 0
    obvVal := obvVal + volume
else if price_change < 0
    obvVal := obvVal - volume
obvEma          = ta.ema(obvVal, obvLength)
volumeUptrend   = obvVal > obvEma
volumeDowntrend = obvVal < obvEma


// ===== DAILY RANGE & MEAN-REV STOP ADJUST =====
dayHigh      = request.security(syminfo.tickerid, "1D", high)
dayLow       = request.security(syminfo.tickerid, "1D", low)
dayRange     = dayHigh - dayLow
rangeUsedPct = ((close - dayLow) / (dayRange == 0 ? 0.0001 : dayRange)) * 100


meanRevDay       = useMeanRevStopAdj and rangeUsedPct > meanRevRangeThresh
float hardStopMultAdj = meanRevDay ? hardStopAtr * meanRevStopFactor : hardStopAtr


var float rangeRemaining      = 1.0
var float rangeRemainingShort = 1.0
if useRangeScaling
    rangeRemaining      := math.max(0.5, (100 - rangeUsedPct) / rangeThresholdPercent)
    rangeRemainingShort := math.max(0.5, rangeUsedPct / rangeThresholdPercent)
else
    rangeRemaining      := 1.0
    rangeRemainingShort := 1.0


// ===== ENTRY CONDITIONS =====
macdBullish = ta.crossover(macdLine, signalLine)
macdBearish = ta.crossunder(macdLine, signalLine)
rsiBullish  = rsi > effectiveRsiBullMinVal and rsi < effectiveRsiBullMaxVal
rsiBearish  = rsi < effectiveRsiBearMaxVal and rsi > effectiveRsiBearMinVal


longConditionBase  = macdBullish and rsiBullish and (not useMTFTrend or htfBull) and volatilityOk and inSession and liquidityOk
shortConditionBase = enableShorts and macdBearish and rsiBearish and (not useMTFTrend or htfBear) and volatilityOk and inSession and liquidityOk


longCondition      = longConditionBase  and gapFilterOk and (not isEventSoon) and (not useVolumeConfirmation or volumeUptrend)  and hourlyVolOk
shortCondition     = shortConditionBase and gapFilterOk and (not isEventSoon) and (not useVolumeConfirmation or volumeDowntrend) and hourlyVolOk


// ===== PINECONNECTOR MESSAGE STRINGS (FIX #3: proper pip conversion) =====
string pcLotsStr = str.tostring(pcLots)

// FIX #3: Convert price distance to PineConnector pips
// PC defines 1 pip = 10 points = syminfo.mintick * 10
float onePcPip       = syminfo.mintick * 10
float slDistPrice    = atr * hardStopMultAdj
float tpDistPrice    = atr * riskRewardRatio
float slDist         = math.round(slDistPrice / onePcPip, 1)
float tpDist         = math.round(tpDistPrice / onePcPip, 1)


// FIX #1 & #2: Build messages — entries use alert_message=, no separate alert()
string pc_buy_msg     = PC_ID + ",buy,"  + PC_SYMBOL + ",vol_lots=" + pcLotsStr
string pc_sell_msg    = PC_ID + ",sell," + PC_SYMBOL + ",vol_lots=" + pcLotsStr
string pc_close_long  = PC_ID + ",closelong,"  + PC_SYMBOL
string pc_close_short = PC_ID + ",closeshort," + PC_SYMBOL

// Atomic reversal messages
string pc_reverse_to_long  = PC_ID + ",closeshortopenlong,"  + PC_SYMBOL + ",vol_lots=" + pcLotsStr
string pc_reverse_to_short = PC_ID + ",closelongopenshort," + PC_SYMBOL + ",vol_lots=" + pcLotsStr


if usePCtp
    string slTpSuffix = ",sl_pips=" + str.tostring(slDist) + ",tp_pips=" + str.tostring(tpDist)
    pc_buy_msg          := pc_buy_msg  + slTpSuffix
    pc_sell_msg         := pc_sell_msg + slTpSuffix
    pc_reverse_to_long  := pc_reverse_to_long  + slTpSuffix
    pc_reverse_to_short := pc_reverse_to_short + slTpSuffix


// ===== RISK MANAGEMENT =====
stopLoss = atr
tp       = atr * riskRewardRatio * sessionMult * (strategy.position_size > 0 ? rangeRemaining : rangeRemainingShort)


// ===== SCALE-IN TRACKING =====
var int   scaleInsLong           = 0
var int   scaleInsShort          = 0
var float initialLongEntryPrice  = 0.0
var float initialShortEntryPrice = 0.0


// ===== TP LEVEL HIT & BAR TRACKING =====
var bool tpLevelHitLong  = false
var bool tpLevelHitShort = false
var int  barsInProfit    = 0
var int  barsInTrade     = 0


// ===== MFE TRACKING =====
var float mfeCurr = 0.0


if strategy.position_size == 0
    tpLevelHitLong         := false
    tpLevelHitShort        := false
    barsInProfit           := 0
    barsInTrade            := 0
    scaleInsLong           := 0
    scaleInsShort          := 0
    initialLongEntryPrice  := 0.0
    initialShortEntryPrice := 0.0
    mfeCurr                := 0.0
else
    barsInTrade := barsInTrade + 1


// Update MFE
if strategy.position_size != 0
    if strategy.position_size > 0
        mfeCurr := math.max(mfeCurr, high - strategy.position_avg_price)
    else
        mfeCurr := math.max(mfeCurr, strategy.position_avg_price - low)


// Track when price reaches TP level
if strategy.position_size > 0 and not tpLevelHitLong and high >= strategy.position_avg_price + tp
    tpLevelHitLong := true
if strategy.position_size < 0 and not tpLevelHitShort and low <= strategy.position_avg_price - tp
    tpLevelHitShort := true


if strategy.position_size > 0 and close > strategy.position_avg_price
    barsInProfit := barsInProfit + 1
else if strategy.position_size < 0 and close < strategy.position_avg_price
    barsInProfit := barsInProfit + 1
else if strategy.position_size != 0
    barsInProfit := 0


// ===== CONDITIONAL TIME STOP (FIX #7: also closes scale-ins) =====
bool timeStopCond = false
if useTimeStop and strategy.position_size != 0 and barsInTrade >= maxBarsInTrade and barsInTrade > 1
    if useCondTimeStop
        volCompression = atr < 0.7 * atrMA
        htfFlipDown   = strategy.position_size > 0 and close < htfEMA
        htfFlipUp     = strategy.position_size < 0 and close > htfEMA
        timeStopCond := (volCompression or htfFlipDown or htfFlipUp)
    else
        timeStopCond := true


if timeStopCond
    if strategy.position_size > 0
        strategy.close("Long",  comment="Time/Cond Stop", alert_message=pc_close_long)
        if useScaleIn
            for i = 1 to maxScaleInsPerTrade
                strategy.close("Long Scale " + str.tostring(i), comment="Time/Cond Stop", alert_message=pc_close_long)
    else
        strategy.close("Short", comment="Time/Cond Stop", alert_message=pc_close_short)
        if useScaleIn
            for i = 1 to maxScaleInsPerTrade

        float lockedGain = mfeCurr * (lockPctOfMFE / 100.0)
        float minTp      = strategy.position_avg_price + lockedGain
        tp := math.max(tp, minTp)
        tp := math.min(tp, newTp)
    if strategy.position_size < 0 and tpLevelHitShort
        float newTp      = strategy.position_avg_price - mfeCurr * runnerMult
        float lockedGain = mfeCurr * (lockPctOfMFE / 100.0)
        float maxTp      = strategy.position_avg_price - lockedGain
        tp := math.min(tp, maxTp)
        tp := math.max(tp, newTp)


// Progressive tightening
if useProgressiveTightening and strategy.position_size != 0
    if barsInTrade > tightenPhase1Bars and barsInTrade <= tightenPhase2Bars
        trailOffset := trailOffset * (1 - tightenPhase1Pct/100)
    else if barsInTrade > tightenPhase2Bars and barsInTrade <= tightenPhase3Bars
        trailOffset := trailOffset * (1 - tightenPhase2Pct/100)
    else if barsInTrade > tightenPhase3Bars
        trailOffset := trailOffset * (1 - tightenPhase3Pct/100)

    trailPoints := na

                  trail_points=useTrailingStop ? trailPoints : na,
                  trail_offset=useTrailingStop ? trailOffset : na,
                  limit=strategy.position_avg_price - tp,
                  stop=useHardStop ? effectiveStop : (not na(breakEvenLevel) ? breakEvenLevel : na),
                  alert_message=pc_close_short)
    if useScaleIn
        for i = 1 to maxScaleInsPerTrade
            strategy.exit("Exit Short Scale " + str.tostring(i), from_entry="Short Scale " + str.tostring(i),
                          trail_points=useTrailingStop ? trailPoints : na,
                          trail_offset=useTrailingStop ? trailOffset : na,
                          limit=strategy.position_avg_price - tp,
                          stop=useHardStop ? effectiveStop : (not na(breakEvenLevel) ? breakEvenLevel : na),
