import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.upper_donchian = self.I(donchian, self.data, 20, 'high')
        self.lower_donchian = self.I(donchian, self.data, 20, 'low')
        self.atr = self.I(atr, self.data, 14)
        self.displacement = np.abs(self.data.Close - self.data.Close.shift(1))

    def _regime_ok(self):
        return self._regime_ok_donchian()

    def _regime_ok_donchian(self):
        return True

    def _filters_ok(self):
        return True

    def _enter_if_signal(self):
        if self.data.Close[-1] > self.upper_donchian[-1] and self.displacement[-1] > 1.2 * self.atr[-1]:
            self.position.enter_long(lots_by_risk_pct(self.spec, self._equity_start, self.data.Close[-1]))
            self.sl_price = self.data.Close[-1] - 100 * self.data.Close[-1] / 100000
            self.tp_price = self.data.Close[-1] + 500 * self.data.Close[-1] / 100000
        elif self.data.Close[-1] < self.lower_donchian[-1] and self.displacement[-1] > 1.2 * self.atr[-1]:
            self.position.enter_short(lots_by_risk_pct(self.spec, self._equity_start, self.data.Close[-1]))
            self.sl_price = self.data.Close[-1] + 100 * self.data.Close[-1] / 100000
            self.tp_price = self.data.Close[-1] - 500 * self.data.Close[-1] / 100000

    def _manage_open(self):
        time_stop = self.spec.get("exit_rules", {}).get("time_stop", {}).get("num_bars")
        if time_stop is not None:
            trade = self.trades[-1] if self.trades else None
            if trade is not None:
                bars_open = len(self.data) - trade.entry_bar
                if bars_open >= time_stop:
                    self.position.close()
