```python
from __future__ import annotations

import json
import os
from typing import Any, Dict, Optional

import numpy as np
import pandas as pd

from agents import signals, regime, risk, config
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    _spec: Dict[str, Any] = {
        "strategy_id": "gen12_xau_asia_london_breakout_m5",
        "symbol": "XAUUSD",
        "timeframe": "M5",
        "risk": {"risk_pct": 0.5, "daily_dd_kill_pct": 3.0},
        "exit": {"time_stop_bars": 20},
    }

    def init(self):
        super().init()

        self._atr_series = self.I(signals.atr, self.data, 14)
        self._adx_series = self.I(regime.adx, self.data, 14)
        self._atr_pct_series = self.I(regime.atr_percentile, self.data, 14, 200)

        # Precompute Asia range per day and trade window mask
        df = self.data.df if hasattr(self.data, "df") else self.data
        idx = pd.DatetimeIndex(df.index)
        if idx.tz is None:
            idx_utc = idx.tz_localize("UTC")
        else:
            idx_utc = idx.tz_convert("UTC")

        hours = idx_utc.hour
        minutes = idx_utc.minute
        hm = hours * 60 + minutes

        asia_start = 0 * 60
        asia_end = 6 * 60  # exclusive
        trade_start = 7 * 60
        trade_end = 10 * 60  # exclusive

        asia_mask = (hm >= asia_start) & (hm < asia_end)
        trade_mask = (hm >= trade_start) & (hm < trade_end)

        self._trade_mask = np.asarray(trade_mask, dtype=bool)

        dates = idx_utc.strftime("%Y-%m-%d").to_numpy()
        self._dates = dates

        high = np.asarray(df["High"].values, dtype=float)
        low = np.asarray(df["Low"].values, dtype=float)

        n = len(df)
        asia_high = np.full(n, np.nan)
        asia_low = np.full(n, np.nan)

        cur_date = None
        cur_high = -np.inf
        cur_low = np.inf
        for i in range(n):
            d = dates[i]
            if d != cur_date:
                cur_date = d
                cur_high = -np.inf
                cur_low = np.inf
            if asia_mask[i]:
                if high[i] > cur_high:
                    cur_high = high[i]
                if low[i] < cur_low:
                    cur_low = low[i]
            # After asia session ends, record running asia range
            if not asia_mask[i] and cur_high != -np.inf:
                asia_high[i] = cur_high
                asia_low[i] = cur_low

        self._asia_high = asia_high
        self._asia_low = asia_low
        self._traded_date: Optional[str] = None
        self._last_trade_date: Optional[str] = None

    def _regime_ok(self) -> bool:
        if len(self._adx_series) < 1:
            return False
        adx_val = float(self._adx_series[-1])
        if np.isnan(adx_val) or adx_val < 18:
            return False
        pct = float(self._atr_pct_series[-1])
        if np.isnan(pct):
            return False
        if pct < 30 or pct > 95:
            return False
        return True

    def _filters_ok(self) -> bool:
        bar_i = len(self.data) - 1
        if bar_i < 0 or bar_i >= len(self._trade_mask):
            return False
        if not bool(self._trade_mask[bar_i]):
            return False
        return True

    def _enter_if_signal(self) -> None:
        if self.position:
            return

        bar_i = len(self.data) - 1
        cur_date = self._dates[bar_i]

        if self._last_trade_date == cur_date:
            return

        atr_val = float(self._atr_series[-1])
        if np.isnan(atr_val) or atr_val <= 0:
            return

        a_hi = self._asia_high[bar_i]
        a_lo = self._asia_low[bar_i]
        if np.isnan(a_hi) or np.isnan(a_lo):
            return

        asia_range = a_hi - a_lo
        if asia_range < 0.5 * atr_val or asia_range > 2.0 * atr_val:
            return

        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)

        if body < 1.0 * atr_val:
            return

        breakout_thr = 0.5 * atr_val

        long_signal = close >= a_hi + breakout_thr and close > open_
        short_signal = close <= a_lo - breakout_thr and close < open_

        if not (long_signal or short_signal):
            return

        risk_pct = 0.5
        equity = float(self.equity)

        if long_signal:
            sl = close - 0.75 * atr_val
            tp = close + 2.0 * atr_val
            stop_dist = close - sl
            if stop_dist <= 0:
                return
            size = risk.lots_by_risk_pct(equity, risk_pct, stop_dist, close)
            if isinstance(size, float) and 0 < size < 1:
                pass
            else:
                size = max(1, int(size))
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.buy(size=size, sl=sl, tp=tp)
                self._last_trade_date = cur_date
            except Exception:
                pass
        elif short_signal:
            sl = close + 0.75 * atr_val
            tp = close - 2.0 * atr_val
            stop_dist = sl - close
            if stop_dist <= 0:
                return
            size = risk.lots_by_risk_pct(equity, risk_pct, stop_dist, close)
            if isinstance(size, float) and 0 < size < 1:
                pass
            else:
                size = max(1, int(size))
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.sell(size=size, sl=sl, tp=tp)
                self._last_trade_date = cur_date
            except Exception:
                pass

    def _manage_open(self) -> None:
        if not self.position:
            return

        time_stop = 20
        if self.trades:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= time_stop:
                self.position.close()
                return

        atr_val = float(self._atr_series[-1])
        if np.isnan(atr_val) or atr_val <= 0:
            return

        price = float(self.data.Close[-1])
        for trade in self.trades:
            entry = trade.entry_price
            if trade.is_long:
                favorable = price - entry
                if favorable >= 1.0 * atr_val:
                    if trade.sl is None or trade.sl < entry:
                        trade.sl = entry
            else:
                favorable = entry - price
                if favorable >= 1.0 * atr_val:
                    if trade.sl is None or trade.sl > entry:
                        trade.sl = entry

    def next(self):
        if not self._filters_ok():
            self._manage_open()
            return
        if not self._regime_ok():
            self._manage_open()
            return
        self._enter_if_signal()
        self._manage_open()
```
