import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self._adx_series = self.I(adx, self.data, self.spec["regime_filter"]["params"]["threshold"])
        self._rsi_series = self.I(rsi, self.data, self.spec["entry_rule"]["params"]["rsi_period"])
        self._atr_series = self.I(atr, self.data, self.spec["exit_rule"]["params"]["trailing_stop_pips"])

    def _regime_ok(self) -> bool:
        return self._adx_series[-1] > self.spec["regime_filter"]["params"]["threshold"]

    def _filters_ok(self) -> bool:
        return True

    def _enter_if_signal(self) -> None:
        if self._regime_ok() and self._filters_ok():
            if self._rsi_series[-1] < self.spec["entry_rule"]["params"]["rsi_thresholds"][0]:
                self.position.enter_long(self.data.Close[-1])
                self.sl_price = self.data.Close[-1] - self.spec["exit_rule"]["params"]["trailing_stop_pips"]
                self.tp_price = self.data.Close[-1] + self.spec["exit_rule"]["params"]["trailing_stop_pips"]
            elif self._rsi_series[-1] > self.spec["entry_rule"]["params"]["rsi_thresholds"][1]:
                self.position.enter_short(self.data.Close[-1])
                self.sl_price = self.data.Close[-1] + self.spec["exit_rule"]["params"]["trailing_stop_pips"]
                self.tp_price = self.data.Close[-1] - self.spec["exit_rule"]["params"]["trailing_stop_pips"]

    def _manage_open(self) -> None:
        if self.position:
            if self.position.is_long:
                self.sl_price = max(self.sl_price, self.data.Close[-1] - self.spec["exit_rule"]["params"]["trailing_stop_pips"])
            else:
                self.sl_price = min(self.sl_price, self.data.Close[-1] + self.spec["exit_rule"]["params"]["trailing_stop_pips"])
