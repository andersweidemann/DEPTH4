import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: dict = {}
    _symbol: str = "XAUUSD"
    _equity_start: float = 10_000.0
    sl_price: float = None
    tp_price: float = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("filters", {}).get("session_utc") or []
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, sessions), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self._bb_series = self.I(bollinger, self.data, self.spec["entry_rule"]["params"]["bb_period"], self.spec["entry_rule"]["params"]["bb_deviation"])
        self._rsi_series = self.I(rsi, self.data, self.spec["entry_rule"]["params"]["rsi_period"])
        self._bb_width_series = self.I(bb_width, self.data, self.spec["regime_filter"]["params"]["period"])
        self._atr_series = self.I(atr, self.data, self.spec["sl_rule"]["params"]["atr_period"])

    def _regime_ok(self):
        bb_width_percentile = self.spec["regime_filter"]["params"]["percentile"]
        bb_width_now = float(self._bb_width_series[-1])
        bb_width_percentile_value = np.percentile(self._bb_width_series, bb_width_percentile)
        return bb_width_now > bb_width_percentile_value

    def _filters_ok(self):
        return super()._filters_ok()

    def _enter_if_signal(self):
        if self.position:
            return
        bb_now = self._bb_series[-1]
        rsi_now = float(self._rsi_series[-1])
        if (bb_now["upper"] <= self.data.Close[-1] and rsi_now >= self.spec["entry_rule"]["params"]["rsi_thresholds"][1]) or \
           (bb_now["lower"] >= self.data.Close[-1] and rsi_now <= self.spec["entry_rule"]["params"]["rsi_thresholds"][0]):
            self.sl_price = self.data.Close[-1] - self.spec["sl_rule"]["params"]["atr_multiplier"] * float(self._atr_series[-1])
            self.tp_price = self.data.Close[-1] + abs(self.data.Close[-1] - bb_now["upper"]) if self.data.Close[-1] < bb_now["upper"] else self.data.Close[-1] - abs(self.data.Close[-1] - bb_now["lower"])
            lots = lots_by_risk_pct(self.spec["sizing_rule"]["params"]["fraction"], self._equity_start, self.data.Close[-1], self.sl_price)
            self.position.enter(lots)

    def _manage_open(self):
        if not self.position:
            return
        if self.data.Close[-1] >= self.tp_price:
            self.position.close()
        elif self.data.Close[-1] <= self.sl_price:
            self.position.close()
        time_stop = self.spec.get("time_stop_rule", {}).get("params", {}).get("bars")
        if time_stop is not None:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= time_stop:
                self.position.close()
