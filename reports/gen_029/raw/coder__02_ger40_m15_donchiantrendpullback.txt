```python
import json
import os
from pathlib import Path
from typing import Any, Dict, Optional

import numpy as np
import pandas as pd

from backtesting import Strategy

from agents import signals, regime, risk, config
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    def init(self):
        super().init()
        p = self.spec.get("params", {})
        self.donchian_period = int(p.get("donchian_period", 20))
        self.ema_fast_p = int(p.get("ema_fast", 20))
        self.ema_slow_p = int(p.get("ema_slow", 50))
        self.atr_period = int(p.get("atr_period", 14))
        self.adx_min = float(p.get("adx_min", 22))
        self.pullback_tol = float(p.get("pullback_tolerance_atr", 0.3))
        self.sl_atr_mult = float(p.get("sl_atr_mult", 1.8))
        self.tp_atr_mult = float(p.get("tp_atr_mult", 3.6))
        self.time_stop_bars = int(p.get("time_stop_bars", 40))
        self.trail_atr_mult = 2.0
        self.breakout_lookback = 30
        self.cooldown_bars = int(self.spec.get("sizing", {}).get("cooldown_bars", 6))
        self.risk_pct = float(self.spec.get("sizing", {}).get("risk_pct", 0.75))

        self._ema_fast = self.I(signals.ema, self.data.Close, self.ema_fast_p)
        self._ema_slow = self.I(signals.ema, self.data.Close, self.ema_slow_p)
        self._atr_series = self.I(signals.atr, self.data, self.atr_period)
        donch = self.I(signals.donchian, self.data, self.donchian_period)
        if isinstance(donch, tuple) or (hasattr(donch, "ndim") and donch.ndim == 2):
            self._donch_high = donch[0]
            self._donch_low = donch[1]
        else:
            self._donch_high = donch
            self._donch_low = donch

        self._adx_series = self.I(regime.adx, self.data, 14)

        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        hours = pd.DatetimeIndex(idx).hour
        self._session_mask_full = (hours >= 7) & (hours < 16)

        self._last_exit_bar = -10_000

    def _regime_ok(self) -> bool:
        i = len(self.data) - 1
        if i < max(self.ema_slow_p, self.atr_period, 14) + 2:
            return False
        adx_v = float(self._adx_series[-1])
        if np.isnan(adx_v) or adx_v < self.adx_min:
            return False
        ef = float(self._ema_fast[-1])
        es = float(self._ema_slow[-1])
        ef_prev = float(self._ema_fast[-3])
        es_prev = float(self._ema_slow[-3])
        aligned_up = ef > es and ef > ef_prev and es > es_prev
        aligned_dn = ef < es and ef < ef_prev and es < es_prev
        return aligned_up or aligned_dn

    def _filters_ok(self) -> bool:
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        return True

    def _had_recent_breakout(self, direction: str) -> bool:
        n = min(self.breakout_lookback, len(self.data) - 2)
        if n <= 0:
            return False
        closes = np.asarray(self.data.Close)[-(n + 1):-1]
        if direction == "long":
            dh = np.asarray(self._donch_high)[-(n + 2):-2]
            if len(dh) != len(closes):
                m = min(len(dh), len(closes))
                return bool(np.any(closes[-m:] > dh[-m:]))
            return bool(np.any(closes > dh))
        else:
            dl = np.asarray(self._donch_low)[-(n + 2):-2]
            if len(dl) != len(closes):
                m = min(len(dl), len(closes))
                return bool(np.any(closes[-m:] < dl[-m:]))
            return bool(np.any(closes < dl))

    def _enter_if_signal(self) -> None:
        if self.position:
            return
        bar_i = len(self.data) - 1
        if bar_i - self._last_exit_bar < self.cooldown_bars:
            return

        close = float(self.data.Close[-1])
        close_prev = float(self.data.Close[-2])
        high = float(self.data.High[-1])
        low = float(self.data.Low[-1])
        ef = float(self._ema_fast[-1])
        es = float(self._ema_slow[-1])
        ef_prev = float(self._ema_fast[-3])
        es_prev = float(self._ema_slow[-3])
        atr_v = float(self._atr_series[-1])
        if np.isnan(atr_v) or atr_v <= 0:
            return

        tol = self.pullback_tol * atr_v

        long_ok = (
            self._had_recent_breakout("long")
            and ef > es and ef > ef_prev and es > es_prev
            and abs(low - ef) <= tol
            and close > ef
            and close > close_prev
        )
        short_ok = (
            self._had_recent_breakout("short")
            and ef < es and ef < ef_prev and es < es_prev
            and abs(high - ef) <= tol
            and close < ef
            and close < close_prev
        )

        if not (long_ok or short_ok):
            return

        if long_ok:
            swing_low = float(np.min(np.asarray(self.data.Low)[-10:]))
            sl_atr = close - self.sl_atr_mult * atr_v
            sl = min(sl_atr, swing_low)
            tp = close + self.tp_atr_mult * atr_v
            if sl >= close:
                return
            risk_per_unit = close - sl
        else:
            swing_high = float(np.max(np.asarray(self.data.High)[-10:]))
            sl_atr = close + self.sl_atr_mult * atr_v
            sl = max(sl_atr, swing_high)
            tp = close - self.tp_atr_mult * atr_v
            if sl <= close:
                return
            risk_per_unit = sl - close

        equity = float(self.equity)
        try:
            size = risk.lots_by_risk_pct(equity, self.risk_pct, risk_per_unit, price=close)
        except TypeError:
            size = risk.lots_by_risk_pct(equity, self.risk_pct, risk_per_unit)

        if size is None or size <= 0:
            return
        if isinstance(size, float) and 0 < size < 1:
            pass
        else:
            size = max(1, int(size))

        self.sl_price = sl
        self.tp_price = tp

        if long_ok:
            self.buy(size=size, sl=sl, tp=tp)
        else:
            self.sell(size=size, sl=sl, tp=tp)

    def _manage_open(self) -> None:
        if not self.position:
            return

        if self.trades:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= self.time_stop_bars:
                self.position.close()
                self._last_exit_bar = len(self.data) - 1
                return

        atr_v = float(self._atr_series[-1])
        if np.isnan(atr_v):
            return
        price = float(self.data.Close[-1])

        for trade in self.trades:
            entry = trade.entry_price
            if trade.is_long:
                r = entry - (trade.sl if trade.sl is not None else entry - atr_v * self.sl_atr_mult)
                if r <= 0:
                    continue
                if price - entry >= r:
                    new_sl = price - self.trail_atr_mult * atr_v
                    if trade.sl is None or new_sl > trade.sl:
                        trade.sl = new_sl
            else:
                r = (trade.sl if trade.sl is not None else entry + atr_v * self.sl_atr_mult) - entry
                if r <= 0:
                    continue
                if entry - price >= r:
                    new_sl = price + self.trail_atr_mult * atr_v
                    if trade.sl is None or new_sl < trade.sl:
                        trade.sl = new_sl

    def next(self):
        if not self._regime_ok():
            self._manage_open()
            return
        if not self._filters_ok():
            self._manage_open()
            return
        self._enter_if_signal()
        self._manage_open()
```
