```python
import json
import os
from typing import Optional

import numpy as np
import pandas as pd

from agents.backtester import RegimeStrategy
from agents import signals, regime, risk


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    atr_period = 14
    min_or_atr = 0.4
    max_or_atr = 2.5
    breakout_body_atr = 0.8
    sl_atr_cap = 1.5
    tp_or_mult = 2.0
    risk_pct = 0.6
    cooldown_bars = 24
    time_stop_bars = 48

    def init(self):
        super().init()
        self._atr_series = self.I(signals.atr, self.data, self.atr_period)
        self._adx_series = self.I(regime.adx, self.data, 14)
        self._atr_pct_series = self.I(regime.atr_percentile, self.data, 100)

        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        self._idx = pd.DatetimeIndex(idx)

        self._or_high = None
        self._or_low = None
        self._or_day = None
        self._breakout_done_day = None
        self._last_exit_bar = -10_000
        self._tp1_hit_trades = set()
        self._or_range_active = None

    def _current_time(self):
        return self._idx[len(self.data) - 1]

    def _in_or_window(self, ts) -> bool:
        t = ts.time()
        return (t >= pd.Timestamp("13:30").time()) and (t < pd.Timestamp("14:00").time())

    def _in_trade_window(self, ts) -> bool:
        t = ts.time()
        return (t >= pd.Timestamp("14:00").time()) and (t < pd.Timestamp("16:00").time())

    def _past_close_time(self, ts) -> bool:
        return ts.time() >= pd.Timestamp("20:00").time()

    def _update_or(self, ts):
        day = ts.date()
        if self._or_day != day:
            self._or_day = day
            self._or_high = None
            self._or_low = None
            self._breakout_done_day = None
            self._or_range_active = None

        if self._in_or_window(ts):
            h = float(self.data.High[-1])
            l = float(self.data.Low[-1])
            if self._or_high is None or h > self._or_high:
                self._or_high = h
            if self._or_low is None or l < self._or_low:
                self._or_low = l

    def _regime_ok(self) -> bool:
        if len(self._atr_pct_series) < 1 or len(self._adx_series) < 1:
            return False
        atr_pct = float(self._atr_pct_series[-1])
        adx_val = float(self._adx_series[-1])
        if np.isnan(atr_pct) or np.isnan(adx_val):
            return False
        if atr_pct < 50:
            return False
        if adx_val < 15:
            return False
        return True

    def _filters_ok(self) -> bool:
        return True

    def next(self):
        ts = self._current_time()
        bar_i = len(self.data) - 1

        self._update_or(ts)
        self._manage_position(ts)

        if self.position:
            return

        if self._past_close_time(ts):
            return

        if not self._in_trade_window(ts):
            return

        if self._or_high is None or self._or_low is None:
            return

        if self._breakout_done_day == ts.date():
            return

        if bar_i - self._last_exit_bar < self.cooldown_bars:
            return

        if not self._regime_ok():
            return

        atr_val = float(self._atr_series[-1])
        if np.isnan(atr_val) or atr_val <= 0:
            return

        or_range = self._or_high - self._or_low
        if or_range <= 0:
            return
        if or_range < self.min_or_atr * atr_val:
            return
        if or_range > self.max_or_atr * atr_val:
            return

        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)
        min_body = self.breakout_body_atr * atr_val

        price = close

        if close > self._or_high and body >= min_body:
            sl_dist = min(or_range, self.sl_atr_cap * atr_val)
            sl = price - sl_dist
            tp = price + self.tp_or_mult * or_range
            tp1 = price + 1.0 * or_range
            self._try_enter_long(price, sl, tp, tp1, or_range)
            self._breakout_done_day = ts.date()
        elif close < self._or_low and body >= min_body:
            sl_dist = min(or_range, self.sl_atr_cap * atr_val)
            sl = price + sl_dist
            tp = price - self.tp_or_mult * or_range
            tp1 = price - 1.0 * or_range
            self._try_enter_short(price, sl, tp, tp1, or_range)
            self._breakout_done_day = ts.date()

    def _position_size(self, price, sl):
        risk_dist = abs(price - sl)
        if risk_dist <= 0:
            return 0
        try:
            size = risk.lots_by_risk_pct(
                equity=self.equity,
                risk_pct=self.risk_pct,
                entry=price,
                stop=sl,
                symbol=self._symbol,
            )
        except Exception:
            size = (self.equity * (self.risk_pct / 100.0)) / risk_dist
        if isinstance(size, float) and 0 < size < 1:
            return size
        try:
            size_int = int(size)
            return max(size_int, 1)
        except Exception:
            return 1

    def _try_enter_long(self, price, sl, tp, tp1, or_range):
        size = self._position_size(price, sl)
        if not size or size <= 0:
            return
        self.sl_price = sl
        self.tp_price = tp
        try:
            self.buy(size=size, sl=sl, tp=tp1)
            self.buy(size=size, sl=sl, tp=tp)
        except Exception:
            try:
                self.buy(sl=sl, tp=tp)
            except Exception:
                return
        self._or_range_active = or_range

    def _try_enter_short(self, price, sl, tp, tp1, or_range):
        size = self._position_size(price, sl)
        if not size or size <= 0:
            return
        self.sl_price = sl
        self.tp_price = tp
        try:
            self.sell(size=size, sl=sl, tp=tp1)
            self.sell(size=size, sl=sl, tp=tp)
        except Exception:
            try:
                self.sell(sl=sl, tp=tp)
            except Exception:
                return
        self._or_range_active = or_range

    def _manage_position(self, ts):
        if not self.position:
            return

        if self._past_close_time(ts):
            self.position.close()
            self._last_exit_bar = len(self.data) - 1
            return

        if self.trades:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= self.time_stop_bars:
                self.position.close()
                self._last_exit_bar = len(self.data) - 1
                return

        atr_val = float(self._atr_series[-1]) if len(self._atr_series) else np.nan
        or_range = self._or_range_active
        if not np.isnan(atr_val) and or_range and self.trades:
            price = float(self.data.Close[-1])
            trail = 1.5 * atr_val
            for trade in self.trades:
                if trade.is_long:
                    if price - trade.entry_price >= or_range:
                        new_sl = price - trail
                        if trade.sl is None or new_sl > trade.sl:
                            trade.sl = new_sl
                else:
                    if trade.entry_price - price >= or_range:
                        new_sl = price + trail
                        if trade.sl is None or new_sl < trade.sl:
                            trade.sl = new_sl
```
