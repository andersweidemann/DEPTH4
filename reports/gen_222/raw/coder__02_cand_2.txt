import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        super().init()
        self._atr_series = self.I(atr, self.data, 14)
        self._sma_series = self.I(sma, self.data, 10)
        self._donchian_series = self.I(donchian, self.data, 10)

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        ind = rf.get("indicator", "atr_percentile")
        if ind == "atr_percentile":
            atr_val = float(self._atr_series[-1])
            percentile = rf.get("params", {}).get("percentile", 50)
            lookback = rf.get("params", {}).get("lookback", 100)
            atr_percentile_val = atr_percentile(self._atr_series, lookback, percentile)
            return atr_val > atr_percentile_val
        return True

    def _filters_ok(self):
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self):
        entry_rule = self.spec.get("entry_rule")
        if entry_rule.get("type") == "asia_london_breakout":
            asia_range_period = entry_rule.get("params", {}).get("asia_range_period", 10)
            london_breakout_threshold = entry_rule.get("params", {}).get("london_breakout_threshold", 1.2)
            if self.data.High[-1] > self._sma_series[-1] * london_breakout_threshold and self.data.Low[-1] < self._sma_series[-1] / london_breakout_threshold:
                self.position.enter(long=True, size=lots_by_risk_pct(self.spec, self.data, self.equity))
                self.sl_price = self.data.Low[-1] - 2 * float(self._atr_series[-1])
                self.tp_price = self.data.High[-1] + 3 * float(self._atr_series[-1])

    def _manage_open(self):
        exit_rule = self.spec.get("exit_rule")
        if exit_rule.get("type") == "opposite_range":
            if self.position.is_long and self.data.Low[-1] < self._donchian_series[-1]:
                self.position.close()
            elif not self.position.is_long and self.data.High[-1] > self._donchian_series[-1]:
                self.position.close()
        time_stop = self.spec.get("time_stop")
        if time_stop is not None:
            trade = self.trades[-1] if self.trades else None
            if trade is not None:
                bars_open = len(self.data) - trade.entry_bar
                if bars_open >= time_stop:
                    self.position.close()
