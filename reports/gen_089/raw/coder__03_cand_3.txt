import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import bb_width, sma, ema, atr, rsi, bollinger, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.upper_bb = self.I(bollinger, self.data, n=20, std_dev=2).upper
        self.lower_bb = self.I(bollinger, self.data, n=20, std_dev=2).lower
        self.bb_width = self.I(bb_width, self.data, n=20)
        self.threshold = 0.5
        self._broker_spread_points = 0

    def _regime_ok(self):
        return self.bb_width[-1] < self.threshold

    def _filters_ok(self):
        return True

    def _enter_if_signal(self):
        if self.spec["entry_rules"]["long"]["condition"] == "close > upper_bb && bb_width < threshold":
            if self.data.Close[-1] > self.upper_bb[-1] and self.bb_width[-1] < self.threshold:
                self.position.enter_long(lots_by_risk_pct(self._equity_start, self.spec["sizing_rules"]["params"]["size"]))
                self.sl_price = self.data.Close[-1] - 150 * self._broker_spread_points
                self.tp_price = self.data.Close[-1] + 600 * self._broker_spread_points
        elif self.spec["entry_rules"]["short"]["condition"] == "close < lower_bb && bb_width < threshold":
            if self.data.Close[-1] < self.lower_bb[-1] and self.bb_width[-1] < self.threshold:
                self.position.enter_short(lots_by_risk_pct(self._equity_start, self.spec["sizing_rules"]["params"]["size"]))
                self.sl_price = self.data.Close[-1] + 150 * self._broker_spread_points
                self.tp_price = self.data.Close[-1] - 600 * self._broker_spread_points

    def _manage_open(self):
        time_stop = self.spec["exit_rules"]["time_stop"]["params"]["num_bars"]
        if self.position:
            if time_stop is not None:
                bars_open = len(self.data) - self.position.entry_bar
                if bars_open >= time_stop:
                    self.position.close()
            else:
                if self.position.is_long:
                    if self.data.Close[-1] < self.sl_price:
                        self.position.close()
                    elif self.data.Close[-1] > self.tp_price:
                        self.position.close()
                elif self.position.is_short:
                    if self.data.Close[-1] > self.sl_price:
                        self.position.close()
                    elif self.data.Close[-1] < self.tp_price:
                        self.position.close()
