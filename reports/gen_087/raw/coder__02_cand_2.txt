import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self._bb_width = self.I(bb_width, self.data, self.spec["regime_filter"]["params"]["min_width"])
        self._bollinger = self.I(bollinger, self.data, self.spec["entry_rule"]["params"]["bb_period"], self.spec["entry_rule"]["params"]["bb_deviation"])
        self._atr = self.I(atr, self.data, self.spec["sl_rule"]["params"]["atr_period"])
        self._sizing_rule = self.spec["sizing_rule"]["params"]["fraction"]

    def _regime_ok(self):
        return self._bb_width[-1] > self.spec["regime_filter"]["params"]["min_width"] and self._bb_width[-1] < self.spec["regime_filter"]["params"]["max_width"]

    def _filters_ok(self):
        return super()._filters_ok()

    def _enter_if_signal(self):
        if self._regime_ok() and self._filters_ok():
            if self._bollinger['lower'][-1] <= self.data.Close[-1] <= self._bollinger['upper'][-1]:
                size = lots_by_risk_pct(self._equity_start, self.spec["sizing_rule"]["params"]["fraction"])
                self.position.enter(size)
                self.sl_price = self.data.Close[-1] - self.spec["sl_rule"]["params"]["atr_multiplier"] * self._atr[-1]
                self.tp_price = self.data.Close[-1] + self.spec["tp_rule"]["params"]["rr"] * (self.data.Close[-1] - self.sl_price)

    def _manage_open(self):
        if self.position:
            if self.data.Close[-1] >= self._bollinger['upper'][-1]:
                self.position.close()
            elif self.spec["time_stop"] is not None and len(self.data) - self.position.entry_bar >= self.spec["time_stop"]:
                self.position.close()
