import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import atr, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("filters", {}).get("session_utc") or []
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, sessions), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self.upper_atr = self.I(atr_breakout_levels, self.data, n=20)[0]
        self.lower_atr = self.I(atr_breakout_levels, self.data, n=20)[1]
        self.breakout = self.I(atr_breakout_levels, self.data, n=20)[2]

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf.get("type") == "atr":
            threshold = rf.get("params", {}).get("threshold")
            atr_val = float(self.I(atr, self.data, n=20)[-1])
            return atr_val < threshold
        return True

    def _filters_ok(self):
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self):
        entry_rules = self.spec.get("entry_rules")
        if entry_rules:
            long_condition = entry_rules.get("long", {}).get("condition")
            short_condition = entry_rules.get("short", {}).get("condition")
            if long_condition and self.breakout > self.upper_atr and pd.Timestamp(self.data.index[-1]).hour >= 9 and pd.Timestamp(self.data.index[-1]).hour < 12:
                self.position.open(long=True, size=lots_by_risk_pct(self.spec.get("sizing_rules", {}).get("params", {}).get("size", 0.1), self.equity, self.data.Close[-1]))
                self.sl_price = self.data.Close[-1] - self.spec.get("exit_rules", {}).get("stop_loss", {}).get("params", {}).get("pips", 50) * self.data.pip
                self.tp_price = self.data.Close[-1] + self.spec.get("exit_rules", {}).get("take_profit", {}).get("params", {}).get("pips", 200) * self.data.pip
            elif short_condition and self.breakout < self.lower_atr and pd.Timestamp(self.data.index[-1]).hour >= 9 and pd.Timestamp(self.data.index[-1]).hour < 12:
                self.position.open(long=False, size=lots_by_risk_pct(self.spec.get("sizing_rules", {}).get("params", {}).get("size", 0.1), self.equity, self.data.Close[-1]))
                self.sl_price = self.data.Close[-1] + self.spec.get("exit_rules", {}).get("stop_loss", {}).get("params", {}).get("pips", 50) * self.data.pip
                self.tp_price = self.data.Close[-1] - self.spec.get("exit_rules", {}).get("take_profit", {}).get("params", {}).get("pips", 200) * self.data.pip

    def _manage_open(self):
        exit_rules = self.spec.get("exit_rules")
        if exit_rules:
            time_stop = exit_rules.get("time_stop")
            if time_stop and self.position:
                trade = self.trades[-1]
                bars_open = len(self.data) - trade.entry_bar
                if bars_open >= time_stop.get("params", {}).get("num_hours", 1) * 60:
                    self.position.close()
            stop_loss = exit_rules.get("stop_loss")
            if stop_loss and self.position:
                if self.position.is_long and self.data.Close[-1] < self.sl_price:
                    self.position.close()
                elif not self.position.is_long and self.data.Close[-1] > self.sl_price:
                    self.position.close()
            take_profit = exit_rules.get("take_profit")
            if take_profit and self.position:
                if self.position.is_long and self.data.Close[-1] > self.tp_price:
                    self.position.close()
                elif not self.position.is_long and self.data.Close[-1] < self.tp_price:
                    self.position.close()
