import json
import os
from pathlib import Path
from typing import Optional

import numpy as np
import pandas as pd

from agents.backtester import RegimeStrategy
from agents import signals, regime, risk


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    def init(self):
        super().init()
        spec = self.spec

        self._atr_series = self.I(signals.atr, self.data, 14)

        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        idx = pd.DatetimeIndex(idx)

        asia_mask = (idx.hour >= 0) & (idx.hour < 6)
        london_mask = (idx.hour >= 7) & (idx.hour < 10)

        self._asia_mask = np.asarray(asia_mask, dtype=bool)
        self._london_mask = np.asarray(london_mask, dtype=bool)
        self._dates = np.array([d.date() for d in idx])

        high = np.asarray(self.data.High, dtype=float)
        low = np.asarray(self.data.Low, dtype=float)

        n = len(idx)
        asia_high = np.full(n, np.nan)
        asia_low = np.full(n, np.nan)
        asia_range = np.full(n, np.nan)

        cur_date = None
        running_high = -np.inf
        running_low = np.inf
        day_high = np.nan
        day_low = np.nan

        for i in range(n):
            d = self._dates[i]
            if d != cur_date:
                cur_date = d
                running_high = -np.inf
                running_low = np.inf
                day_high = np.nan
                day_low = np.nan
            if self._asia_mask[i]:
                if high[i] > running_high:
                    running_high = high[i]
                if low[i] < running_low:
                    running_low = low[i]
                day_high = running_high
                day_low = running_low
            asia_high[i] = day_high
            asia_low[i] = day_low
            if not np.isnan(day_high) and not np.isnan(day_low):
                asia_range[i] = day_high - day_low

        self._asia_high = asia_high
        self._asia_low = asia_low
        self._asia_range = asia_range

        london_open_bar = np.full(n, -1, dtype=int)
        cur_date = None
        opened = False
        open_bar = -1
        for i in range(n):
            d = self._dates[i]
            if d != cur_date:
                cur_date = d
                opened = False
                open_bar = -1
            if self._london_mask[i] and not opened:
                opened = True
                open_bar = i
            london_open_bar[i] = open_bar
        self._london_open_bar = london_open_bar

        atr_vals = np.asarray(self._atr_series, dtype=float)
        self._atr_pct = np.full(n, np.nan)
        lookback = 500
        for i in range(n):
            lo = max(0, i - lookback + 1)
            window = atr_vals[lo:i + 1]
            window = window[~np.isnan(window)]
            if len(window) > 20:
                cur = atr_vals[i]
                if not np.isnan(cur):
                    self._atr_pct[i] = (window < cur).sum() / len(window) * 100.0

        self._last_entry_bar = -10_000
        self._breakout_done_date = None

    def _regime_ok(self) -> bool:
        i = len(self.data) - 1
        atr_v = float(self._atr_series[-1])
        if np.isnan(atr_v) or atr_v <= 0:
            return False
        rng = self._asia_range[i]
        if np.isnan(rng):
            return False
        ratio = rng / atr_v
        if ratio < 0.5 or ratio > 2.0:
            return False
        pct = self._atr_pct[i]
        if np.isnan(pct) or pct <= 25:
            return False
        return True

    def _filters_ok(self) -> bool:
        i = len(self.data) - 1
        if not self._london_mask[i]:
            return False
        return True

    def _enter_if_signal(self) -> None:
        if self.position:
            return
        i = len(self.data) - 1
        bar_idx = i - self._london_open_bar[i]
        if self._london_open_bar[i] < 0 or bar_idx > 9:
            return

        if i - self._last_entry_bar < 12:
            return

        today = self._dates[i]
        if self._breakout_done_date == today:
            return

        atr_v = float(self._atr_series[-1])
        if np.isnan(atr_v) or atr_v <= 0:
            return

        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)
        if body < 1.2 * atr_v:
            return

        ah = self._asia_high[i]
        al = self._asia_low[i]
        if np.isnan(ah) or np.isnan(al):
            return

        long_trigger = ah + 0.5 * atr_v
        short_trigger = al - 0.5 * atr_v

        direction = 0
        if close > long_trigger:
            direction = 1
        elif close < short_trigger:
            direction = -1
        else:
            return

        if direction == 1:
            sl_range = al
            sl_atr = close - 1.5 * atr_v
            sl = max(sl_range, sl_atr)
            if sl >= close:
                return
            tp = close + 2.5 * atr_v
        else:
            sl_range = ah
            sl_atr = close + 1.5 * atr_v
            sl = min(sl_range, sl_atr)
            if sl <= close:
                return
            tp = close - 2.5 * atr_v

        risk_dist = abs(close - sl)
        if risk_dist <= 0:
            return

        equity = float(self.equity)
        size = risk.lots_by_risk_pct(equity, 0.5, risk_dist, 1.0)
        try:
            size = float(size)
        except Exception:
            size = 0.0
        if size <= 0:
            frac = (0.005 * equity) / (risk_dist * 1.0)
            size = max(frac / equity, 0.001)
            size = min(size, 0.99)
            if size <= 0 or size >= 1:
                return

        self.sl_price = sl
        self.tp_price = tp

        try:
            if direction == 1:
                self.buy(size=size, sl=sl, tp=tp)
            else:
                self.sell(size=size, sl=sl, tp=tp)
            self._last_entry_bar = i
            self._breakout_done_date = today
        except Exception:
            return

    def _manage_open(self) -> None:
        if not self.position or not self.trades:
            return

        i = len(self.data) - 1
        time_stop = 48
        trade = self.trades[-1]
        bars_open = len(self.data) - trade.entry_bar
        if bars_open >= time_stop:
            self.position.close()
            return

        atr_v = float(self._atr_series[-1])
        if np.isnan(atr_v):
            return
        price = float(self.data.Close[-1])

        for tr in self.trades:
            entry = tr.entry_price
            if tr.is_long:
                init_risk = entry - (tr.sl if tr.sl is not None else entry)
                if init_risk <= 0:
                    continue
                if price - entry >= init_risk:
                    new_sl = price - 3.0 * atr_v
                    if tr.sl is None or new_sl > tr.sl:
                        tr.sl = new_sl
            else:
                init_risk = (tr.sl if tr.sl is not None else entry) - entry
                if init_risk <= 0:
                    continue
                if entry - price >= init_risk:
                    new_sl = price + 3.0 * atr_v
                    if tr.sl is None or new_sl < tr.sl:
                        tr.sl = new_sl

    def next(self):
        if not self._filters_ok():
            self._manage_open()
            return
        if not self._regime_ok():
            self._manage_open()
            return
        self._enter_if_signal()
        self._manage_open()
