```python
import json
import os
from pathlib import Path
from typing import Any, Dict, Optional

import numpy as np
import pandas as pd

from agents import signals, regime, risk, config
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    def init(self):
        super().init()
        spec_file = Path(__file__).parent / self.spec_path
        if spec_file.exists():
            try:
                with open(spec_file, "r") as f:
                    self.spec = json.load(f)
            except Exception:
                pass

        self._atr_series = self.I(signals.atr, self.data, 14)
        self._adx_series = self.I(regime.adx, self.data, 14)
        self._atr_pct_series = self.I(regime.atr_percentile, self.data, 14, 200)

        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        self._asia_mask = np.asarray(
            signals.session_mask(idx, ["00:00-06:00"]), dtype=bool)
        self._london_mask = np.asarray(
            signals.session_mask(idx, ["07:00-10:00"]), dtype=bool)

        self._asia_high = np.nan
        self._asia_low = np.nan
        self._asia_day = None
        self._london_start_bar = -1
        self._traded_day = None
        self._be_moved_trade = None

    def _regime_ok(self) -> bool:
        i = len(self.data) - 1
        if i < 200:
            return False
        atr_pct = float(self._atr_pct_series[-1])
        if np.isnan(atr_pct):
            return False
        if not (0.3 <= atr_pct <= 0.9):
            return False
        adx_val = float(self._adx_series[-1])
        if np.isnan(adx_val) or adx_val < 15:
            return False
        return True

    def _filters_ok(self) -> bool:
        i = len(self.data) - 1
        if 0 <= i < len(self._london_mask):
            if not bool(self._london_mask[i]):
                return False
        return True

    def _update_asia_range(self):
        i = len(self.data) - 1
        ts = pd.Timestamp(self.data.index[i])
        today = ts.normalize()

        if bool(self._asia_mask[i]):
            if self._asia_day != today:
                self._asia_day = today
                self._asia_high = float(self.data.High[-1])
                self._asia_low = float(self.data.Low[-1])
            else:
                self._asia_high = max(self._asia_high, float(self.data.High[-1]))
                self._asia_low = min(self._asia_low, float(self.data.Low[-1]))

        if bool(self._london_mask[i]):
            prev_in_london = i > 0 and bool(self._london_mask[i - 1])
            if not prev_in_london:
                self._london_start_bar = i

    def next(self):
        self._update_asia_range()

        self._manage_open()

        if self.position:
            return

        if not self._regime_ok():
            return
        if not self._filters_ok():
            return

        self._enter_if_signal()

    def _enter_if_signal(self):
        i = len(self.data) - 1
        ts = pd.Timestamp(self.data.index[i])
        today = ts.normalize()

        if self._traded_day == today:
            return
        if np.isnan(self._asia_high) or np.isnan(self._asia_low):
            return
        if self._asia_day != today:
            return
        if self._london_start_bar < 0:
            return
        if i - self._london_start_bar >= 24:
            return

        atr_val = float(self._atr_series[-1])
        if np.isnan(atr_val) or atr_val <= 0:
            return

        rng = self._asia_high - self._asia_low
        if not (0.5 * atr_val <= rng <= 2.0 * atr_val):
            return

        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)

        long_trigger = self._asia_high + 0.5 * atr_val
        short_trigger = self._asia_low - 0.5 * atr_val

        equity = float(self.equity)
        risk_pct = float(self.spec.get("sizing", {}).get("risk_per_trade_pct", 0.75))

        if close > long_trigger and body >= 1.0 * atr_val:
            sl = close - 0.75 * atr_val
            tp = close + 2.25 * atr_val
            sl_dist = close - sl
            if sl_dist <= 0:
                return
            units = risk.lots_by_risk_pct(equity, risk_pct, sl_dist, self._symbol)
            units = max(1, int(units))
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.buy(size=units, sl=sl, tp=tp)
                self._traded_day = today
                self._be_moved_trade = None
            except Exception:
                pass

        elif close < short_trigger and body >= 1.0 * atr_val:
            sl = close + 0.75 * atr_val
            tp = close - 2.25 * atr_val
            sl_dist = sl - close
            if sl_dist <= 0:
                return
            units = risk.lots_by_risk_pct(equity, risk_pct, sl_dist, self._symbol)
            units = max(1, int(units))
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.sell(size=units, sl=sl, tp=tp)
                self._traded_day = today
                self._be_moved_trade = None
            except Exception:
                pass

    def _manage_open(self):
        if not self.position or not self.trades:
            return

        i = len(self.data) - 1
        ts = pd.Timestamp(self.data.index[i])
        atr_val = float(self._atr_series[-1]) if not np.isnan(self._atr_series[-1]) else 0.0
        price = float(self.data.Close[-1])

        for trade in list(self.trades):
            bars_open = i - trade.entry_bar
            if bars_open >= 36:
                trade.close()
                continue
            if ts.hour >= 15:
                trade.close()
                continue

            if atr_val > 0:
                entry = trade.entry_price
                if trade.is_long:
                    if price - entry >= 1.0 * atr_val:
                        if trade.sl is None or trade.sl < entry:
                            trade.sl = entry
                else:
                    if entry - price >= 1.0 * atr_val:
                        if trade.sl is None or trade.sl > entry:
                            trade.sl = entry
```
