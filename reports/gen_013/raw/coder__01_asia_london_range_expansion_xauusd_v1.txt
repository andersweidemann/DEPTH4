```python
import json
import os
from pathlib import Path
from typing import Optional

import numpy as np
import pandas as pd

from agents.backtester import RegimeStrategy
from agents import signals, regime, risk


class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"

    def init(self):
        super().init()
        spec_file = Path(__file__).parent / self.spec_path
        if spec_file.exists():
            try:
                with open(spec_file, "r") as f:
                    self.spec.update(json.load(f))
            except Exception:
                pass

        self._atr_series = self.I(signals.atr, self.data, 14)

        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        idx = pd.DatetimeIndex(idx)
        if idx.tz is None:
            idx_utc = idx.tz_localize("UTC")
        else:
            idx_utc = idx.tz_convert("UTC")

        hours = idx_utc.hour.values
        minutes = idx_utc.minute.values
        dates = idx_utc.strftime("%Y-%m-%d").values

        minutes_of_day = hours * 60 + minutes

        self._asia_mask = (minutes_of_day >= 0) & (minutes_of_day < 6 * 60)
        self._london_mask = (minutes_of_day >= 7 * 60) & (minutes_of_day < 10 * 60)
        self._eod_mask = (minutes_of_day >= 15 * 60)
        self._dates = dates
        self._minutes_of_day = minutes_of_day

        high = np.asarray(self.data.High)
        low = np.asarray(self.data.Low)
        n = len(high)

        asia_high = np.full(n, np.nan)
        asia_low = np.full(n, np.nan)

        cur_date = None
        cur_high = -np.inf
        cur_low = np.inf
        have_asia = False

        for i in range(n):
            d = dates[i]
            if d != cur_date:
                cur_date = d
                cur_high = -np.inf
                cur_low = np.inf
                have_asia = False
            if self._asia_mask[i]:
                if high[i] > cur_high:
                    cur_high = high[i]
                if low[i] < cur_low:
                    cur_low = low[i]
                have_asia = True
            if have_asia and minutes_of_day[i] >= 6 * 60:
                asia_high[i] = cur_high
                asia_low[i] = cur_low

        self._asia_high = self.I(lambda: asia_high, name="asia_high")
        self._asia_low = self.I(lambda: asia_low, name="asia_low")

        self._atr_pct = self.I(regime.atr_percentile, self.data, 14, 200)

        self._broker_spread_points = 0
        self._last_breakout_date: Optional[str] = None

    def _regime_ok(self) -> bool:
        v = float(self._atr_pct[-1]) if len(self._atr_pct) else np.nan
        if np.isnan(v):
            return False
        return 25.0 <= v <= 90.0

    def _filters_ok(self) -> bool:
        bar_i = len(self.data) - 1
        if not self._london_mask[bar_i]:
            return False
        return True

    def next(self):
        bar_i = len(self.data) - 1
        today = self._dates[bar_i]

        if self.position:
            if self._eod_mask[bar_i]:
                self.position.close()
                return

            atr_now = float(self._atr_series[-1])
            if not np.isnan(atr_now) and self.trades:
                price = float(self.data.Close[-1])
                for trade in self.trades:
                    entry = trade.entry_price
                    if trade.is_long:
                        if price - entry >= 1.0 * atr_now:
                            if trade.sl is None or entry > trade.sl:
                                trade.sl = entry
                    else:
                        if entry - price >= 1.0 * atr_now:
                            if trade.sl is None or entry < trade.sl:
                                trade.sl = entry
            return

        if not self._regime_ok():
            return
        if not self._filters_ok():
            return

        atr_now = float(self._atr_series[-1])
        if np.isnan(atr_now) or atr_now <= 0:
            return

        ah = float(self._asia_high[-1])
        al = float(self._asia_low[-1])
        if np.isnan(ah) or np.isnan(al):
            return

        rng = ah - al
        if not (0.5 * atr_now <= rng <= 2.0 * atr_now):
            return

        if self._last_breakout_date == today:
            return

        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)

        if body < 1.0 * atr_now:
            return

        equity = self.equity
        risk_pct = 0.5

        long_trigger = ah + 0.5 * atr_now
        short_trigger = al - 0.5 * atr_now

        if close > long_trigger:
            sl = close - 0.75 * atr_now
            tp = close + 2.25 * atr_now
            sl_dist = close - sl
            if sl_dist <= 0:
                return
            units = risk.lots_by_risk_pct(equity, risk_pct, sl_dist, self._symbol)
            if units is None or units <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.buy(size=units, sl=sl, tp=tp)
                self._last_breakout_date = today
            except Exception:
                pass

        elif close < short_trigger:
            sl = close + 0.75 * atr_now
            tp = close - 2.25 * atr_now
            sl_dist = sl - close
            if sl_dist <= 0:
                return
            units = risk.lots_by_risk_pct(equity, risk_pct, sl_dist, self._symbol)
            if units is None or units <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.sell(size=units, sl=sl, tp=tp)
                self._last_breakout_date = today
            except Exception:
                pass
```
