import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("regime_filter", {}).get("params", {}).get("start_hour", None)
        if sessions is not None:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, [(sessions, 10)]), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self._high_series = self.data.High
        self._low_series = self.data.Low
        self._close_series = self.data.Close
        self._donchian_high = self.I(donchian, self.data, 10, 'high')
        self._donchian_low = self.I(donchian, self.data, 10, 'low')
        self._atr = self.I(atr, self.data, 10)

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        start_hour = rf.get("params", {}).get("start_hour")
        end_hour = rf.get("params", {}).get("end_hour")
        current_hour = pd.Timestamp(self.data.index[-1]).hour
        return start_hour <= current_hour <= end_hour

    def _filters_ok(self):
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self):
        if self.position:
            return
        if not self._regime_ok() or not self._filters_ok():
            return
        breakout_threshold = self.spec.get("entry_rule", {}).get("params", {}).get("breakout_threshold")
        consolidation_period = self.spec.get("entry_rule", {}).get("params", {}).get("consolidation_period")
        if self._high_series[-1] > self._donchian_high[-1] * breakout_threshold:
            self.sl_price = self._donchian_low[-1]
            self.tp_price = self._high_series[-1] + (self._high_series[-1] - self._donchian_low[-1]) * 2
            size = lots_by_risk_pct(self._spec, self.equity, self._broker_spread_points, self.sl_price, self.tp_price)
            self.position.enter(long=True, size=size)
        elif self._low_series[-1] < self._donchian_low[-1] / breakout_threshold:
            self.sl_price = self._donchian_high[-1]
            self.tp_price = self._low_series[-1] - (self._donchian_high[-1] - self._low_series[-1]) * 2
            size = lots_by_risk_pct(self._spec, self.equity, self._broker_spread_points, self.sl_price, self.tp_price)
            self.position.enter(long=False, size=size)

    def _manage_open(self):
        exit_cfg = self.spec.get("exit_rule")
        time_stop = exit_cfg.get("params", {}).get("time_stop")
        if not self.position:
            return
        trade = self.trades[-1]
        bars_open = len(self.data) - trade.entry_bar
        if time_stop is not None and bars_open >= time_stop:
            self.position.close()
            return
        trailing_stop = exit_cfg.get("params", {}).get("trailing_stop")
        trailing_step = exit_cfg.get("params", {}).get("trailing_step")
        if trailing_stop is not None and trailing_step is not None:
            if self.position.is_long:
                new_sl = self._close_series[-1] - trailing_stop
                if trade.sl is None or new_sl > trade.sl:
                    trade.sl = new_sl
            else:
                new_sl = self._close_series[-1] + trailing_stop
                if trade.sl is None or new_sl < trade.sl:
                    trade.sl = new_sl
