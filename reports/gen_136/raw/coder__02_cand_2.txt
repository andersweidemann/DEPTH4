import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import bollinger, bb_width, atr
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.bollinger_bands = self.I(bollinger, self.data, self.spec["entry_rule"]["params"]["bb_period"], self.spec["entry_rule"]["params"]["bb_deviation"])
        self.bb_width = self.I(bb_width, self.data, self.spec["entry_rule"]["params"]["bb_period"])
        self.atr = self.I(atr, self.data, 14)
        self._broker_spread_points = 0

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        min_width = rf["params"]["min_width"]
        if self.bb_width[-1] < min_width:
            return False
        return True

    def _filters_ok(self):
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.1)):
            return False
        return True

    def _enter_if_signal(self):
        bb_period = self.spec["entry_rule"]["params"]["bb_period"]
        bb_deviation = self.spec["entry_rule"]["params"]["bb_deviation"]
        if self.data.Close[-1] <= self.bollinger_bands[-1][0] and self.position.size == 0:
            self.position.enter_long(lots_by_risk_pct(self.spec["sizing_rule"]["params"]["fraction"], self.atr[-1]))
            self.sl_price = self.data.Close[-1] - self.spec["exit_rule"]["params"]["sl"] * self.atr[-1]
            self.tp_price = self.bollinger_bands[-1][1]
        elif self.data.Close[-1] >= self.bollinger_bands[-1][1] and self.position.size == 0:
            self.position.enter_short(lots_by_risk_pct(self.spec["sizing_rule"]["params"]["fraction"], self.atr[-1]))
            self.sl_price = self.data.Close[-1] + self.spec["exit_rule"]["params"]["sl"] * self.atr[-1]
            self.tp_price = self.bollinger_bands[-1][0]

    def _manage_open(self):
        time_stop = self.spec["exit_rule"]["params"]["time_stop"]
        if self.position:
            bars_open = len(self.data) - self.position.entry_bar
            if bars_open >= time_stop:
                self.position.close()
        if self.position and self.position.is_long and self.data.Close[-1] >= self.tp_price:
            self.position.close()
        elif self.position and not self.position.is_long and self.data.Close[-1] <= self.tp_price:
            self.position.close()
