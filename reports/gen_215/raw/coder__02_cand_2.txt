import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import rsi, atr
from agents.risk import lots_by_risk_pct, DailyKillState, daily_kill_ok, spread_ok

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: dict = {}
    _symbol: str = "BTCUSD"
    _equity_start: float = 10_000.0
    sl_price: float = None
    tp_price: float = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.rsi_series = self.I(rsi, self.data, 7)
        self.atr_series = self.I(atr, self.data, 14)
        self.atr_percentile_series = np.percentile(self.atr_series, 50)

    def _regime_ok(self) -> bool:
        return self.atr_series[-1] < self.atr_percentile_series

    def _filters_ok(self) -> bool:
        return True

    def _enter_if_signal(self) -> None:
        if self.rsi_series[-1] < 20 and not self.position:
            self.sl_price = self.data.Close[-1] - 200
            self.tp_price = self.data.Close[-1] + 400
            self.position.enter_long(lots_by_risk_pct(self.spec, self.equity, 0.05))

        elif self.rsi_series[-1] > 80 and not self.position:
            self.sl_price = self.data.Close[-1] + 200
            self.tp_price = self.data.Close[-1] - 400
            self.position.enter_short(lots_by_risk_pct(self.spec, self.equity, 0.05))

    def _manage_open(self) -> None:
        if self.position:
            if len(self.data) - self.position.entry_bar >= 15:
                self.position.close()
