import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: Dict[str, Any] = {}
    _symbol: str = "XAUUSD"
    _equity_start: float = 10_000.0
    sl_price: Optional[float] = None
    tp_price: Optional[float] = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("regime_filter", {}).get("params", {}).get("asia_session")
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, sessions), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self.high_asia = self.I(donchian, self.data, self.spec["regime_filter"]["params"]["start_hour"], self.spec["regime_filter"]["params"]["end_hour"], 'high')
        self.low_asia = self.I(donchian, self.data, self.spec["regime_filter"]["params"]["start_hour"], self.spec["regime_filter"]["params"]["end_hour"], 'low')
        self.rsi = self.I(rsi, self.data, 7)
        self.atr = self.I(atr, self.data, 14)

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf["type"] == "asia_session":
            return True
        return super()._regime_ok()

    def _filters_ok(self):
        return super()._filters_ok()

    def _enter_if_signal(self):
        if self.spec["entry_rules"]["long"]["condition"] == "close > high_asia && rsi(7) < 30":
            if self.data.Close[-1] > self.high_asia[-1] and self.rsi[-1] < 30:
                self.position.enter_long()
                self.sl_price = self.data.Close[-1] - self.spec["exit_rules"]["sl"]["params"]["multiplier"] * self.atr[-1]
                self.tp_price = self.data.Close[-1] + (self.data.Close[-1] - self.low_asia[-1])
        elif self.spec["entry_rules"]["short"]["condition"] == "close < low_asia && rsi(7) > 70":
            if self.data.Close[-1] < self.low_asia[-1] and self.rsi[-1] > 70:
                self.position.enter_short()
                self.sl_price = self.data.Close[-1] + self.spec["exit_rules"]["sl"]["params"]["multiplier"] * self.atr[-1]
                self.tp_price = self.data.Close[-1] - (self.high_asia[-1] - self.data.Close[-1])

    def _manage_open(self):
        exit_cfg = self.spec.get("exit_rules", {})
        time_stop = exit_cfg.get("time_stop", {}).get("params", {}).get("num_bars")
        if not self.position:
            return
        if time_stop is not None:
            trade = self.trades[-1] if self.trades else None
            if trade is not None:
                bars_open = len(self.data) - trade.entry_bar
                if bars_open >= time_stop:
                    self.position.close()
                    return
        trail_mult = exit_cfg.get("sl", {}).get("params", {}).get("multiplier")
        if trail_mult and hasattr(self, "_atr_series") and self.trades:
            atr_now = float(self._atr_series[-1])
            if not np.isnan(atr_now):
                price = float(self.data.Close[-1])
                for trade in self.trades:
                    if trade.is_long and trade.pl_pct > 0:
                        new_sl = price - trail_mult * atr_now
                        if trade.sl is None or new_sl > trade.sl:
                            trade.sl = new_sl
                    elif not trade.is_long and trade.pl_pct > 0:
                        new_sl = price + trail_mult * atr_now
                        if trade.sl is None or new_sl < trade.sl:
                            trade.sl = new_sl
