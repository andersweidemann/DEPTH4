import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: Dict[str, Any] = {}
    _symbol: str = "XAUUSD"
    _equity_start: float = 10_000.0
    sl_price: Optional[float] = None
    tp_price: Optional[float] = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.upper_bb, self.lower_bb = self.I(bollinger, self.data.Close, 20, 2)
        self.rsi = self.I(rsi, self.data.Close, 7)
        self.bb_width = self.I(bb_width, self.data.Close, 20, 2)
        self._regime_series = self.I(adx, self.data.High, self.data.Low, self.data.Close, 14)
        self._atr_series = self.I(atr, self.data.High, self.data.Low, self.data.Close, 14)

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf["type"] == "bb_width_percentile":
            percentile = rf["params"]["percentile"]
            lookback = rf["params"]["lookback"]
            bb_width_series = self.I(bb_width, self.data.Close, 20, 2)
            bb_width_percentile = np.percentile(bb_width_series[-lookback:], percentile)
            return self.bb_width[-1] > bb_width_percentile
        return True

    def _filters_ok(self):
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.1)):
            return False
        return True

    def _enter_if_signal(self):
        entry_rules = self.spec.get("entry_rules")
        if entry_rules:
            long_condition = entry_rules["long"]["condition"]
            short_condition = entry_rules["short"]["condition"]
            if long_condition == "close < lower_bb && rsi(7) < 10":
                if self.data.Close[-1] < self.lower_bb[-1] and self.rsi[-1] < 10:
                    self.position.enter_long(lots_by_risk_pct(self.spec, self.data, self.equity))
                    self.sl_price = self.data.Close[-1] - self.I(atr, self.data.High, self.data.Low, self.data.Close, 14)[-1] * self.spec["exit_rules"]["sl"]["params"]["multiplier"]
                    self.tp_price = self.upper_bb[-1]
            elif short_condition == "close > upper_bb && rsi(7) > 90":
                if self.data.Close[-1] > self.upper_bb[-1] and self.rsi[-1] > 90:
                    self.position.enter_short(lots_by_risk_pct(self.spec, self.data, self.equity))
                    self.sl_price = self.data.Close[-1] + self.I(atr, self.data.High, self.data.Low, self.data.Close, 14)[-1] * self.spec["exit_rules"]["sl"]["params"]["multiplier"]
                    self.tp_price = self.lower_bb[-1]

    def _manage_open(self):
        exit_rules = self.spec.get("exit_rules")
        if exit_rules:
            time_stop = exit_rules.get("time_stop")
            if time_stop:
                num_bars = time_stop["params"]["num_bars"]
                if self.position:
                    trade = self.trades[-1]
                    bars_open = len(self.data) - trade.entry_bar
                    if bars_open >= num_bars:
                        self.position.close()
            tp = exit_rules.get("tp")
            if tp:
                if tp["type"] == "opposite_bb":
                    if self.position.is_long and self.data.Close[-1] >= self.upper_bb[-1]:
                        self.position.close()
                    elif self.position.is_short and self.data.Close[-1] <= self.lower_bb[-1]:
                        self.position.close()
            sl = exit_rules.get("sl")
            if sl:
                if sl["type"] == "atr":
                    multiplier = sl["params"]["multiplier"]
                    atr = self.I(atr, self.data.High, self.data.Low, self.data.Close, 14)[-1]
                    if self.position.is_long and self.data.Close[-1] <= self.sl_price:
                        self.position.close()
                    elif self.position.is_short and self.data.Close[-1] >= self.sl_price:
                        self.position.close()
