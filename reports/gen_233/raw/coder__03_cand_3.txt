import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("filters", {}).get("session_utc") or []
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, sessions), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self._atr_series = self.I(atr, self.data, 20)
        self._rsi_series = self.I(rsi, self.data, 7)
        self._bb_series = self.I(bb_width, self.data, 20)
        self._bb_deviation = 1.75
        self._rsi_thresholds = [10, 90]

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf.get("type") == "atr_percentile":
            threshold = rf.get("params", {}).get("threshold", 50)
            atr_now = float(self._atr_series[-1])
            return atr_now > np.percentile(self._atr_series, threshold)
        return True

    def _filters_ok(self):
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 10)):
            return False
        return True

    def _enter_if_signal(self):
        er = self.spec.get("entry_rule")
        if er.get("type") == "mean_reversion":
            bb_period = er.get("params", {}).get("bb_period", 20)
            bb_deviation = er.get("params", {}).get("bb_deviation", 1.75)
            rsi_period = er.get("params", {}).get("rsi_period", 7)
            rsi_thresholds = er.get("params", {}).get("rsi_thresholds", [10, 90])
            close = self.data.Close[-1]
            bb_upper = self.data.Close[-bb_period-1:-1].mean() + bb_deviation * self.data.Close[-bb_period-1:-1].std()
            bb_lower = self.data.Close[-bb_period-1:-1].mean() - bb_deviation * self.data.Close[-bb_period-1:-1].std()
            rsi_now = float(self._rsi_series[-1])
            if (close < bb_lower and rsi_now < rsi_thresholds[0]) or (close > bb_upper and rsi_now > rsi_thresholds[1]):
                self.sl_price = close - (bb_upper - bb_lower) * 0.5 if close < bb_lower else close + (bb_upper - bb_lower) * 0.5
                self.tp_price = close + (bb_upper - bb_lower) * 2 if close < bb_lower else close - (bb_upper - bb_lower) * 2
                size = self.spec.get("sizing_rule", {}).get("params", {}).get("size", 0.1)
                self.position.open(size)

    def _manage_open(self):
        exit_cfg = self.spec.get("exit_rule")
        time_stop = exit_cfg.get("params", {}).get("time_stop_bars")
        if not self.position:
            return
        if time_stop is not None:
            trade = self.trades[-1] if self.trades else None
            if trade is not None:
                bars_open = len(self.data) - trade.entry_bar
                if bars_open >= time_stop:
                    self.position.close()
                    return
        tp_pips = exit_cfg.get("params", {}).get("tp_pips", 500)
        sl_pips = exit_cfg.get("params", {}).get("sl_pips", 100)
        if self.position:
            close = self.data.Close[-1]
            if self.position.is_long and close >= self.position.entry_price + tp_pips * self._symbol.pip:
                self.position.close()
            elif not self.position.is_long and close <= self.position.entry_price - tp_pips * self._symbol.pip:
                self.position.close()
            elif self.position.is_long and close <= self.position.entry_price - sl_pips * self._symbol.pip:
                self.position.close()
            elif not self.position.is_long and close >= self.position.entry_price + sl_pips * self._symbol.pip:
                self.position.close()
