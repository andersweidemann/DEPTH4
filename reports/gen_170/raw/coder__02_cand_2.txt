import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import bollinger, bb_width, rsi, sma
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.bollinger_bands = self.I(bollinger, self.data, n=self.spec["regime_filter"]["params"]["bb_period"], dev=self.spec["regime_filter"]["params"]["bb_dev"])
        self.bb_width = self.I(bb_width, self.data, n=self.spec["regime_filter"]["params"]["bb_period"])
        self.rsi = self.I(rsi, self.data, n=self.spec["entry_rule"]["params"]["rsi_period"])
        self._broker_spread_points = 0

    def _regime_ok(self):
        min_width = self.spec["regime_filter"]["params"]["min_width"]
        return self.bb_width[-1] > min_width

    def _filters_ok(self):
        return True

    def _enter_if_signal(self):
        rsi_thresholds = self.spec["entry_rule"]["params"]["rsi_thresholds"]
        if self.rsi[-1] < rsi_thresholds[0] and self.data.Close[-1] <= self.bollinger_bands[-1, 0]:
            lots = lots_by_risk_pct(self.spec["sizing_rule"]["params"]["proportion"], self._equity_start, self.data)
            self.position.enter_long(lots)
            self.sl_price = self.data.Close[-1] - self.spec["exit_rule"]["params"]["trail_pips"] * self.data._pip
            self.tp_price = self.data.Close[-1] + self.spec["exit_rule"]["params"]["trail_pips"] * self.data._pip
        elif self.rsi[-1] > rsi_thresholds[1] and self.data.Close[-1] >= self.bollinger_bands[-1, 2]:
            lots = lots_by_risk_pct(self.spec["sizing_rule"]["params"]["proportion"], self._equity_start, self.data)
            self.position.enter_short(lots)
            self.sl_price = self.data.Close[-1] + self.spec["exit_rule"]["params"]["trail_pips"] * self.data._pip
            self.tp_price = self.data.Close[-1] - self.spec["exit_rule"]["params"]["trail_pips"] * self.data._pip

    def _manage_open(self):
        trail_pips = self.spec["exit_rule"]["params"]["trail_pips"]
        step_pips = self.spec["exit_rule"]["params"]["step_pips"]
        for trade in self.trades:
            if trade.is_long and trade.pl_pct > 0:
                new_sl = self.data.Close[-1] - trail_pips * self.data._pip
                if trade.sl is None or new_sl > trade.sl:
                    trade.sl = new_sl
            elif not trade.is_long and trade.pl_pct > 0:
                new_sl = self.data.Close[-1] + trail_pips * self.data._pip
                if trade.sl is None or new_sl < trade.sl:
                    trade.sl = new_sl
