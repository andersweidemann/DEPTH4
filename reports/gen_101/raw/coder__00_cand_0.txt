import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: Dict[str, Any] = {}
    _symbol: str = "XAUUSD"
    _equity_start: float = 10_000.0
    sl_price: Optional[float] = None
    tp_price: Optional[float] = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("filters", {}).get("session_utc") or []
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, sessions), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self._bb_series = self.I(bollinger, self.data, self.spec["regime_filter"]["params"]["bb_period"], self.spec["regime_filter"]["params"]["bb_deviation"])
        self._rsi_series = self.I(rsi, self.data, self.spec["entry_rule"]["params"]["rsi_period"])
        self._bb_width_series = self.I(bb_width, self.data, self.spec["regime_filter"]["params"]["bb_period"], self.spec["regime_filter"]["params"]["bb_deviation"])

    def _regime_ok(self):
        bb_width = float(self._bb_width_series[-1])
        percentile = self.spec["regime_filter"]["params"]["percentile"]
        bb_width_percentile = np.percentile(self._bb_width_series, percentile)
        return bb_width < bb_width_percentile

    def _filters_ok(self):
        return super()._filters_ok()

    def _enter_if_signal(self):
        rsi = float(self._rsi_series[-1])
        upper_bb = float(self._bb_series[-1, 2])
        lower_bb = float(self._bb_series[-1, 0])
        price = float(self.data.Close[-1])
        if (price >= upper_bb and rsi >= self.spec["entry_rule"]["params"]["rsi_thresholds"][1]) or (price <= lower_bb and rsi <= self.spec["entry_rule"]["params"]["rsi_thresholds"][0]):
            lots = lots_by_risk_pct(self.spec["sizing_rule"]["params"]["fraction"], self._equity_start, self.data)
            self.position.enter(lots)
            self.sl_price = price - self.spec["exit_rule"]["params"]["sl_multiplier"] * (upper_bb - lower_bb) if price >= upper_bb else price + self.spec["exit_rule"]["params"]["sl_multiplier"] * (upper_bb - lower_bb)
            self.tp_price = price + (upper_bb - lower_bb) if price <= lower_bb else price - (upper_bb - lower_bb)

    def _manage_open(self):
        time_stop = self.spec["exit_rule"]["params"]["time_stop_bars"]
        if not self.position:
            return
        trade = self.trades[-1]
        bars_open = len(self.data) - trade.entry_bar
        if bars_open >= time_stop:
            self.position.close()
            return
        upper_bb = float(self._bb_series[-1, 2])
        lower_bb = float(self._bb_series[-1, 0])
        price = float(self.data.Close[-1])
        if self.position.is_long and price >= upper_bb:
            self.position.close()
        elif not self.position.is_long and price <= lower_bb:
            self.position.close()
