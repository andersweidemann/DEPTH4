```python
import json
from pathlib import Path
from typing import Optional, Dict, Any

import numpy as np
import pandas as pd

from agents.backtester import RegimeStrategy
from agents import signals, regime, risk


class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"

    breakout_atr_buffer = 0.3
    body_atr_min = 1.0
    range_atr_min = 0.5
    range_atr_max = 2.0
    sl_atr_mult = 0.75
    tp_R = 2.5
    atr_period = 14
    atr_pct_lookback = 288
    atr_pct_min = 30.0
    atr_pct_max = 95.0
    adx_period = 14
    adx_min = 15.0
    trail_atr_mult = 2.0
    trail_trigger_R = 1.5
    be_trigger_R = 1.0
    time_stop_bars = 48
    risk_pct = 0.6

    def init(self):
        try:
            super().init()
        except Exception:
            self.spec = dict(getattr(self, "_spec", {}) or {})
            self._kill_state = risk.DailyKillState(start_of_day_equity=self._equity_start)
            self._session_mask_full = None
            self._broker_spread_points = 0

        self._atr_series = self.I(signals.atr, self.data, self.atr_period)
        self._adx_series = self.I(regime.adx, self.data, self.adx_period)
        self._atr_pct_series = self.I(
            regime.atr_percentile, self.data, self.atr_period, self.atr_pct_lookback
        )

        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        self._trade_mask = np.asarray(
            signals.session_mask(idx, ["07:00-11:00"]), dtype=bool
        )
        self._asia_mask = np.asarray(
            signals.session_mask(idx, ["00:00-06:00"]), dtype=bool
        )

        self._last_trade_day = None
        self._asia_cache_day = None
        self._asia_high = np.nan
        self._asia_low = np.nan
        self._entry_risk = None
        self._be_moved = False
        self._trail_armed = False

    def _compute_asia_range(self, bar_i: int) -> tuple:
        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        current_ts = pd.Timestamp(idx[bar_i])
        current_day = current_ts.date()
        if self._asia_cache_day == current_day:
            return self._asia_high, self._asia_low

        highs = np.asarray(self.data.High)
        lows = np.asarray(self.data.Low)

        start = max(0, bar_i - 300)
        hi = -np.inf
        lo = np.inf
        found = False
        for i in range(start, bar_i + 1):
            ts = pd.Timestamp(idx[i])
            if ts.date() != current_day:
                continue
            if not self._asia_mask[i]:
                continue
            if highs[i] > hi:
                hi = highs[i]
            if lows[i] < lo:
                lo = lows[i]
            found = True

        self._asia_cache_day = current_day
        if found:
            self._asia_high = hi
            self._asia_low = lo
        else:
            self._asia_high = np.nan
            self._asia_low = np.nan
        return self._asia_high, self._asia_low

    def _regime_ok(self) -> bool:
        adx_val = float(self._adx_series[-1])
        if np.isnan(adx_val) or adx_val < self.adx_min:
            return False
        pct = float(self._atr_pct_series[-1])
        if np.isnan(pct):
            return False
        if pct < self.atr_pct_min or pct > self.atr_pct_max:
            return False
        return True

    def _filters_ok(self) -> bool:
        bar_i = len(self.data) - 1
        if bar_i < 0 or bar_i >= len(self._trade_mask):
            return False
        if not bool(self._trade_mask[bar_i]):
            return False
        idx = self.data.index
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        try:
            if not risk.daily_kill_ok(
                self._kill_state, now_date, self.equity,
                self.spec.get("risk", {}).get("daily_dd_kill_pct", 5.0)
            ):
                return False
        except Exception:
            pass
        return True

    def _force_flat_check(self) -> bool:
        idx = self.data.index
        ts = pd.Timestamp(idx[-1])
        if ts.hour >= 15:
            if self.position:
                self.position.close()
            return True
        return False

    def next(self):
        if len(self.data) < max(self.atr_period, self.adx_period, self.atr_pct_lookback) + 5:
            return

        if self._force_flat_check():
            return

        self._manage_open()

        if self.position:
            return

        if not self._regime_ok():
            return
        if not self._filters_ok():
            return

        bar_i = len(self.data) - 1
        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        current_day = pd.Timestamp(idx[bar_i]).date()

        if self._last_trade_day == current_day:
            return

        atr_now = float(self._atr_series[-1])
        if np.isnan(atr_now) or atr_now <= 0:
            return

        asia_high, asia_low = self._compute_asia_range(bar_i)
        if np.isnan(asia_high) or np.isnan(asia_low):
            return

        rng = asia_high - asia_low
        if rng < self.range_atr_min * atr_now or rng > self.range_atr_max * atr_now:
            return

        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)

        if body < self.body_atr_min * atr_now:
            return

        long_trigger = asia_high + self.breakout_atr_buffer * atr_now
        short_trigger = asia_low - self.breakout_atr_buffer * atr_now

        direction = 0
        if close > long_trigger:
            direction = 1
        elif close < short_trigger:
            direction = -1
        else:
            return

        sl_dist = self.sl_atr_mult * atr_now
        if sl_dist <= 0:
            return

        if direction > 0:
            self.sl_price = close - sl_dist
            self.tp_price = close + self.tp_R * sl_dist
        else:
            self.sl_price = close + sl_dist
            self.tp_price = close - self.tp_R * sl_dist

        try:
            size = risk.lots_by_risk_pct(
                equity=self.equity,
                risk_pct=self.risk_pct,
                entry=close,
                stop=self.sl_price,
                symbol=self._symbol,
            )
        except Exception:
            size = None

        if size is None or (isinstance(size, float) and (np.isnan(size) or size <= 0)):
            risk_cash = self.equity * (self.risk_pct / 100.0)
            dollar_risk_per_unit = abs(close - self.sl_price)
            if dollar_risk_per_unit <= 0:
                return
            units = risk_cash / dollar_risk_per_unit
            size = max(1, int(units))

        if isinstance(size, float):
            if 0 < size < 1:
                pass
            else:
                size = max(1, int(size))

        try:
            if direction > 0:
                self.buy(size=size, sl=self.sl_price, tp=self.tp_price)
            else:
                self.sell(size=size, sl=self.sl_price, tp=self.tp_price)
            self._last_trade_day = current_day
            self._entry_risk = sl_dist
            self._be_moved = False
            self._trail_armed = False
        except Exception:
            return

    def _manage_open(self) -> None:
        if not self.position or not self.trades:
            return

        bars_open = len(self.data) - self.trades[-1].entry_bar
        if bars_open >= self.time_stop_bars:
            self.position.close()
            return

        atr_now = float(self._atr_series[-1])
        if np.isnan(atr_now):
            return

        price = float(self.data.Close[-1])
        risk_dist = self._entry_risk if self._entry_risk else self.sl_atr_mult * atr_now
        if risk_dist <= 0:
            return

        for trade in self.trades:
            entry = trade.entry_price
            if trade.is_long:
                r_mult = (price - entry) / risk_dist
                if r_mult >= self.be_trigger_R and not self._be_moved:
                    if trade.sl is None or entry > trade.sl:
                        trade.sl = entry
                        self._be_moved = True
                if r_mult >= self.trail_trigger_R:
                    new_sl = price - self.trail_atr_mult * atr_now
                    if trade.sl is None or new_sl > trade.sl:
                        trade.sl = new_sl
                    self._trail_armed = True
            else:
                r_mult = (entry - price) / risk_dist
                if r_mult >= self.be_trigger_R and not self._be_moved:
                    if trade.sl is None or entry < trade.sl:
                        trade.sl = entry
                        self._be_moved = True
                if r_mult >= self.trail_trigger_R:
                    new_sl = price + self.trail_atr_mult * atr_now
                    if trade.sl is None or new_sl < trade.sl:
                        trade.sl = new_sl
                    self._trail_armed = True
```
