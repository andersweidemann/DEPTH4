from __future__ import annotations

import json
from pathlib import Path
from typing import Optional

import numpy as np
import pandas as pd

from agents.backtester import RegimeStrategy
from agents import signals, regime, risk


def _asia_high(high: np.ndarray, low: np.ndarray, close: np.ndarray,
               hours: np.ndarray, minutes: np.ndarray,
               day_id: np.ndarray, asia_start_min: int, asia_end_min: int) -> np.ndarray:
    n = len(high)
    out = np.full(n, np.nan)
    cur_day = -1
    cur_high = -np.inf
    finalized_high = np.nan
    for i in range(n):
        tmin = int(hours[i]) * 60 + int(minutes[i])
        d = int(day_id[i])
        if d != cur_day:
            cur_day = d
            cur_high = -np.inf
            finalized_high = np.nan
        if asia_start_min <= tmin < asia_end_min:
            if high[i] > cur_high:
                cur_high = high[i]
        if tmin >= asia_end_min and not np.isinf(cur_high):
            finalized_high = cur_high
        out[i] = finalized_high
    return out


def _asia_low(high: np.ndarray, low: np.ndarray, close: np.ndarray,
              hours: np.ndarray, minutes: np.ndarray,
              day_id: np.ndarray, asia_start_min: int, asia_end_min: int) -> np.ndarray:
    n = len(low)
    out = np.full(n, np.nan)
    cur_day = -1
    cur_low = np.inf
    finalized_low = np.nan
    for i in range(n):
        tmin = int(hours[i]) * 60 + int(minutes[i])
        d = int(day_id[i])
        if d != cur_day:
            cur_day = d
            cur_low = np.inf
            finalized_low = np.nan
        if asia_start_min <= tmin < asia_end_min:
            if low[i] < cur_low:
                cur_low = low[i]
        if tmin >= asia_end_min and not np.isinf(cur_low):
            finalized_low = cur_low
        out[i] = finalized_low
    return out


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    ASIA_START_MIN = 0
    ASIA_END_MIN = 6 * 60
    TRADE_START_MIN = 7 * 60
    TRADE_END_MIN = 10 * 60
    FLAT_MIN = 12 * 60

    ATR_PERIOD = 14
    ADX_PERIOD = 14
    ATR_PCTILE_PERIOD = 100
    ADX_MIN = 18
    ATR_PCTILE_MIN = 0.3
    ATR_PCTILE_MAX = 0.95
    ASIA_RANGE_ATR_MIN = 0.5
    ASIA_RANGE_ATR_MAX = 2.0

    MULT_ENTRY = 0.5
    MULT_BODY = 1.2
    SL_ATR_MULT = 0.75
    TP_ATR_MULT = 2.5
    TRAIL_ATR_MULT = 2.0
    TRAIL_ACTIVATE_R = 1.5
    TIME_STOP_BARS = 36
    RISK_PCT = 0.5

    def init(self):
        try:
            p = Path(__file__).parent / self.spec_path
            if p.exists():
                self._spec = json.loads(p.read_text())
        except Exception:
            self._spec = {}
        super().init()

        df = self.data.df if hasattr(self.data, "df") else None
        idx = df.index if df is not None else self.data.index
        ts = pd.DatetimeIndex(idx)
        if ts.tz is None:
            ts_utc = ts.tz_localize("UTC")
        else:
            ts_utc = ts.tz_convert("UTC")

        hours = ts_utc.hour.to_numpy()
        minutes = ts_utc.minute.to_numpy()
        day_id = (ts_utc.year * 10000 + ts_utc.month * 100 + ts_utc.day).to_numpy()

        self._hours = hours
        self._minutes = minutes
        self._day_id = day_id

        self._atr_series = self.I(signals.atr, self.data, self.ATR_PERIOD)

        self._asia_high = self.I(
            _asia_high,
            self.data.High, self.data.Low, self.data.Close,
            hours, minutes, day_id,
            self.ASIA_START_MIN, self.ASIA_END_MIN,
        )
        self._asia_low = self.I(
            _asia_low,
            self.data.High, self.data.Low, self.data.Close,
            hours, minutes, day_id,
            self.ASIA_START_MIN, self.ASIA_END_MIN,
        )

        self._adx_series = self.I(regime.adx, self.data, self.ADX_PERIOD)
        self._atr_pctile = self.I(regime.atr_percentile, self.data,
                                  self.ATR_PERIOD, self.ATR_PCTILE_PERIOD)

        self._breakout_done_day = -1
        self._entry_r = None

    def _in_trade_window(self) -> bool:
        bar_i = len(self.data) - 1
        tmin = int(self._hours[bar_i]) * 60 + int(self._minutes[bar_i])
        return self.TRADE_START_MIN <= tmin < self.TRADE_END_MIN

    def _past_flat(self) -> bool:
        bar_i = len(self.data) - 1
        tmin = int(self._hours[bar_i]) * 60 + int(self._minutes[bar_i])
        return tmin >= self.FLAT_MIN

    def _regime_ok(self) -> bool:
        adx_v = float(self._adx_series[-1])
        if np.isnan(adx_v) or adx_v < self.ADX_MIN:
            return False
        p = float(self._atr_pctile[-1])
        if np.isnan(p) or p < self.ATR_PCTILE_MIN or p > self.ATR_PCTILE_MAX:
            return False
        atr_v = float(self._atr_series[-1])
        ah = float(self._asia_high[-1])
        al = float(self._asia_low[-1])
        if np.isnan(atr_v) or np.isnan(ah) or np.isnan(al) or atr_v <= 0:
            return False
        asia_range = ah - al
        ratio = asia_range / atr_v
        if ratio < self.ASIA_RANGE_ATR_MIN or ratio > self.ASIA_RANGE_ATR_MAX:
            return False
        return True

    def _filters_ok(self) -> bool:
        return self._in_trade_window()

    def next(self):
        bar_i = len(self.data) - 1
        day = int(self._day_id[bar_i])

        if self._past_flat() and self.position:
            self.position.close()
            return

        self._manage_open()

        if self.position:
            return

        if not self._filters_ok():
            return

        if self._breakout_done_day == day:
            return

        if not self._regime_ok():
            return

        atr_v = float(self._atr_series[-1])
        ah = float(self._asia_high[-1])
        al = float(self._asia_low[-1])
        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)

        if body < self.MULT_BODY * atr_v:
            return

        long_ok = close >= ah + self.MULT_ENTRY * atr_v
        short_ok = close <= al - self.MULT_ENTRY * atr_v

        if not (long_ok or short_ok):
            return

        equity = float(self.equity)

        if long_ok:
            sl = close - self.SL_ATR_MULT * atr_v
            sl = max(sl, al)
            if sl >= close:
                return
            tp = close + self.TP_ATR_MULT * atr_v
            risk_per_unit = close - sl
            size = risk.lots_by_risk_pct(equity, self.RISK_PCT, risk_per_unit, close)
            if size <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            self._entry_r = risk_per_unit
            try:
                self.buy(size=size, sl=sl, tp=tp)
            except Exception:
                try:
                    self.buy(sl=sl, tp=tp)
                except Exception:
                    return
            self._breakout_done_day = day
        elif short_ok:
            sl = close + self.SL_ATR_MULT * atr_v
            sl = min(sl, ah)
            if sl <= close:
                return
            tp = close - self.TP_ATR_MULT * atr_v
            risk_per_unit = sl - close
            size = risk.lots_by_risk_pct(equity, self.RISK_PCT, risk_per_unit, close)
            if size <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            self._entry_r = risk_per_unit
            try:
                self.sell(size=size, sl=sl, tp=tp)
            except Exception:
                try:
                    self.sell(sl=sl, tp=tp)
                except Exception:
                    return
            self._breakout_done_day = day

    def _manage_open(self):
        if not self.position or not self.trades:
            return

        bar_i = len(self.data) - 1
        trade = self.trades[-1]
        bars_open = bar_i - trade.entry_bar
        if bars_open >= self.TIME_STOP_BARS:
            self.position.close()
            return

        atr_v = float(self._atr_series[-1])
        if np.isnan(atr_v) or self._entry_r is None or self._entry_r <= 0:
            return

        price = float(self.data.Close[-1])
        for t in self.trades:
            if t.is_long:
                move = price - t.entry_price
                if move >= self.TRAIL_ACTIVATE_R * self._entry_r:
                    new_sl = price - self.TRAIL_ATR_MULT * atr_v
                    if t.sl is None or new_sl > t.sl:
                        try:
                            t.sl = new_sl
                        except Exception:
                            pass
            else:
                move = t.entry_price - price
                if move >= self.TRAIL_ACTIVATE_R * self._entry_r:
                    new_sl = price + self.TRAIL_ATR_MULT * atr_v
                    if t.sl is None or new_sl < t.sl:
                        try:
                            t.sl = new_sl
                        except Exception:
                            pass
