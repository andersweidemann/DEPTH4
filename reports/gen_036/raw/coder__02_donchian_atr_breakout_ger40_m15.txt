```python
import json
import os
from pathlib import Path
from typing import Any, Dict, Optional

import numpy as np
import pandas as pd

from agents import signals, regime, risk
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    donchian_len = 20
    body_atr_mult = 0.8
    ema_fast_len = 20
    ema_slow_len = 50
    atr_sl_mult = 1.8
    tp_atr_mult = 2.5
    atr_len = 14
    adx_len = 14
    time_stop_bars = 20
    cooldown_bars = 6
    risk_pct = 0.6
    be_atr_mult = 1.0
    trail_trigger_atr = 1.5
    trail_atr_mult = 2.0
    atr_pct_len = 200
    atr_pct_min = 30.0
    atr_pct_max = 95.0
    adx_min = 20.0
    session_start_hour = 7
    session_end_hour = 16

    def init(self):
        spec_file = Path(__file__).parent / self.spec_path
        if spec_file.exists():
            try:
                self._spec = json.loads(spec_file.read_text())
            except Exception:
                self._spec = {}

        super().init()

        self._atr_series = self.I(signals.atr, self.data, self.atr_len)
        dh, dl = self.I(signals.donchian, self.data, self.donchian_len)
        self._don_high = dh
        self._don_low = dl
        self._ema_fast = self.I(signals.ema, self.data.Close, self.ema_fast_len)
        self._ema_slow = self.I(signals.ema, self.data.Close, self.ema_slow_len)
        self._adx_series = self.I(regime.adx, self.data, self.adx_len)
        self._atr_pct_series = self.I(
            regime.atr_percentile, self.data, self.atr_len, self.atr_pct_len
        )

        self._last_exit_bar = -10_000

    def _in_session(self) -> bool:
        ts = pd.Timestamp(self.data.index[-1])
        h = ts.hour
        return self.session_start_hour <= h < self.session_end_hour

    def _regime_ok(self) -> bool:
        adx_v = float(self._adx_series[-1])
        if np.isnan(adx_v) or adx_v < self.adx_min:
            return False
        pct = float(self._atr_pct_series[-1])
        if np.isnan(pct):
            return False
        if pct < self.atr_pct_min or pct > self.atr_pct_max:
            return False
        return True

    def _filters_ok(self) -> bool:
        if not self._in_session():
            return False
        now_date = pd.Timestamp(self.data.index[-1]).strftime("%Y-%m-%d")
        try:
            from agents import config as _config
            daily_kill_pct = self.spec.get("risk", {}).get(
                "daily_dd_kill_pct", _config.load()["risk"]["daily_dd_kill_pct"]
            )
        except Exception:
            daily_kill_pct = self.spec.get("risk", {}).get("daily_dd_kill_pct", 5.0)
        if not risk.daily_kill_ok(self._kill_state, now_date, self.equity, daily_kill_pct):
            return False
        return True

    def _enter_if_signal(self) -> None:
        if self.position:
            return
        bar_i = len(self.data) - 1
        if bar_i - self._last_exit_bar < self.cooldown_bars:
            return

        atr_v = float(self._atr_series[-1])
        if np.isnan(atr_v) or atr_v <= 0:
            return

        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)

        prev_don_high = float(self._don_high[-2]) if len(self._don_high) >= 2 else np.nan
        prev_don_low = float(self._don_low[-2]) if len(self._don_low) >= 2 else np.nan
        if np.isnan(prev_don_high) or np.isnan(prev_don_low):
            return

        ema_f = float(self._ema_fast[-1])
        ema_s = float(self._ema_slow[-1])
        if np.isnan(ema_f) or np.isnan(ema_s):
            return

        body_ok = body > self.body_atr_mult * atr_v

        long_sig = (close > prev_don_high) and body_ok and (ema_f > ema_s)
        short_sig = (close < prev_don_low) and body_ok and (ema_f < ema_s)

        if not (long_sig or short_sig):
            return

        high_entry = float(self.data.High[-1])
        low_entry = float(self.data.Low[-1])

        if long_sig:
            sl = low_entry - self.atr_sl_mult * atr_v
            tp = close + self.tp_atr_mult * atr_v
            stop_dist = close - sl
            if stop_dist <= 0:
                return
            try:
                size = risk.lots_by_risk_pct(
                    equity=self.equity,
                    risk_pct=self.risk_pct,
                    stop_distance=stop_dist,
                    price=close,
                    symbol=self._symbol,
                )
            except Exception:
                size = None
            if size is None or size <= 0:
                frac = (self.risk_pct / 100.0) * self.equity / (stop_dist * 1.0)
                size = max(0.0, min(0.99, frac / self.equity))
                if size <= 0:
                    return
            self.sl_price = sl
            self.tp_price = tp
            try:
                if isinstance(size, float) and 0 < size < 1:
                    self.buy(sl=sl, tp=tp, size=size)
                else:
                    self.buy(sl=sl, tp=tp, size=max(1, int(size)))
            except Exception:
                self.buy(sl=sl, tp=tp)
        elif short_sig:
            sl = high_entry + self.atr_sl_mult * atr_v
            tp = close - self.tp_atr_mult * atr_v
            stop_dist = sl - close
            if stop_dist <= 0:
                return
            try:
                size = risk.lots_by_risk_pct(
                    equity=self.equity,
                    risk_pct=self.risk_pct,
                    stop_distance=stop_dist,
                    price=close,
                    symbol=self._symbol,
                )
            except Exception:
                size = None
            if size is None or size <= 0:
                frac = (self.risk_pct / 100.0) * self.equity / (stop_dist * 1.0)
                size = max(0.0, min(0.99, frac / self.equity))
                if size <= 0:
                    return
            self.sl_price = sl
            self.tp_price = tp
            try:
                if isinstance(size, float) and 0 < size < 1:
                    self.sell(sl=sl, tp=tp, size=size)
                else:
                    self.sell(sl=sl, tp=tp, size=max(1, int(size)))
            except Exception:
                self.sell(sl=sl, tp=tp)

    def _manage_open(self) -> None:
        if not self.position:
            return

        atr_v = float(self._atr_series[-1]) if len(self._atr_series) else np.nan
        price = float(self.data.Close[-1])

        if not np.isnan(atr_v) and self.trades:
            for trade in self.trades:
                entry = trade.entry_price
                if trade.is_long:
                    move = price - entry
                    if move >= self.be_atr_mult * atr_v:
                        if trade.sl is None or trade.sl < entry:
                            trade.sl = entry
                    if move >= self.trail_trigger_atr * atr_v:
                        new_sl = price - self.trail_atr_mult * atr_v
                        if trade.sl is None or new_sl > trade.sl:
                            trade.sl = new_sl
                else:
                    move = entry - price
                    if move >= self.be_atr_mult * atr_v:
                        if trade.sl is None or trade.sl > entry:
                            trade.sl = entry
                    if move >= self.trail_trigger_atr * atr_v:
                        new_sl = price + self.trail_atr_mult * atr_v
                        if trade.sl is None or new_sl < trade.sl:
                            trade.sl = new_sl

        if self.trades:
            trade = self.trades[-1]
            bars_open = len(self.data) - 1 - trade.entry_bar
            if bars_open >= self.time_stop_bars:
                self.position.close()
                self._last_exit_bar = len(self.data) - 1
                return

    def next(self):
        if not self.position and self.trades == [] and hasattr(self, "_prev_pos"):
            if self._prev_pos:
                self._last_exit_bar = len(self.data) - 1
        self._prev_pos = bool(self.position)

        if self.position:
            self._manage_open()
            return

        if not self._regime_ok():
            return
        if not self._filters_ok():
            return
        self._enter_if_signal()
```
