import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import atr, donchian, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: dict = {}
    _symbol: str = "US500"
    _equity_start: float = 10_000.0
    sl_price: float = None
    tp_price: float = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("regime_filter", {}).get("sessions", [])
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, sessions), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self.atr = self.I(atr, self.data, 14)
        self.donchian = self.I(donchian, self.data, 14)

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf["type"] == "asia_range_atr":
            min_atr = rf["params"]["min_atr"]
            max_atr = rf["params"]["max_atr"]
            atr_val = float(self.atr[-1])
            return min_atr <= atr_val <= max_atr
        return True

    def _filters_ok(self):
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct")):
            return False
        return True

    def _enter_if_signal(self):
        entry_rules = self.spec.get("entry_rules")
        if entry_rules:
            long_condition = entry_rules["long"]["condition"]
            short_condition = entry_rules["short"]["condition"]
            upper_range = self.donchian[-1][1]
            lower_range = self.donchian[-1][0]
            breakout = self.data.Close[-1]
            displacement = abs(breakout - (upper_range + lower_range) / 2)
            atr_val = self.atr[-1]
            if long_condition and breakout > upper_range and displacement > 1.2 * atr_val:
                self.position.enter_long()
                self.sl_price = breakout - self.spec["exit_rules"]["sl"]["params"]["pips"]
                self.tp_price = breakout + self.spec["exit_rules"]["tp"]["params"]["pips"]
            elif short_condition and breakout < lower_range and displacement > 1.2 * atr_val:
                self.position.enter_short()
                self.sl_price = breakout + self.spec["exit_rules"]["sl"]["params"]["pips"]
                self.tp_price = breakout - self.spec["exit_rules"]["tp"]["params"]["pips"]

    def _manage_open(self):
        exit_rules = self.spec.get("exit_rules")
        if exit_rules:
            time_stop = exit_rules.get("time_stop", {}).get("bars")
            if time_stop is not None:
                trade = self.trades[-1] if self.trades else None
                if trade is not None:
                    bars_open = len(self.data) - trade.entry_bar
                    if bars_open >= time_stop:
                        self.position.close()
            tp = exit_rules.get("tp", {}).get("pips")
            sl = exit_rules.get("sl", {}).get("pips")
            if tp is not None and sl is not None:
                if self.position.is_long:
                    if self.data.Close[-1] >= self.tp_price:
                        self.position.close()
                    elif self.data.Close[-1] <= self.sl_price:
                        self.position.close()
                elif self.position.is_short:
                    if self.data.Close[-1] <= self.tp_price:
                        self.position.close()
                    elif self.data.Close[-1] >= self.sl_price:
                        self.position.close()
