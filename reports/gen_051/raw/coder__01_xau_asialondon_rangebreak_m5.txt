```python
import json
import os
from pathlib import Path
from typing import Optional

import numpy as np
import pandas as pd

from agents import signals, regime, risk
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    def init(self):
        super().init()

        self._atr_series = self.I(signals.atr, self.data, 14)
        self._adx_series = self.I(regime.adx, self.data, 14)

        df = self.data.df if hasattr(self.data, "df") else self.data
        idx = df.index

        self._asia_mask = np.asarray(
            signals.session_mask(idx, [["00:00", "06:00"]]), dtype=bool
        )
        self._london_mask = np.asarray(
            signals.session_mask(idx, [["07:00", "12:00"]]), dtype=bool
        )

        ts = pd.DatetimeIndex(idx)
        self._dates = np.array([t.date() for t in ts])
        self._hours = ts.hour.values
        self._minutes = ts.minute.values

        high = np.asarray(df["High"].values, dtype=float)
        low = np.asarray(df["Low"].values, dtype=float)

        n = len(df)
        asia_high = np.full(n, np.nan)
        asia_low = np.full(n, np.nan)
        london_bar_idx = np.full(n, -1, dtype=int)

        cur_date = None
        ah = -np.inf
        al = np.inf
        have_asia = False
        london_start_bar = -1

        for i in range(n):
            d = self._dates[i]
            if d != cur_date:
                cur_date = d
                ah = -np.inf
                al = np.inf
                have_asia = False
                london_start_bar = -1

            if self._asia_mask[i]:
                if high[i] > ah:
                    ah = high[i]
                if low[i] < al:
                    al = low[i]
                have_asia = True

            if self._london_mask[i]:
                if london_start_bar < 0:
                    london_start_bar = i

            if have_asia:
                asia_high[i] = ah
                asia_low[i] = al
            london_bar_idx[i] = london_start_bar

        self._asia_high = asia_high
        self._asia_low = asia_low
        self._london_start = london_bar_idx

        self._traded_dates = set()

    def _regime_ok(self) -> bool:
        if len(self._adx_series) == 0:
            return False
        v = float(self._adx_series[-1])
        if np.isnan(v):
            return False
        return v >= 15.0

    def _filters_ok(self) -> bool:
        i = len(self.data) - 1
        if i < 0 or i >= len(self._london_mask):
            return False
        return bool(self._london_mask[i])

    def _enter_if_signal(self) -> None:
        if self.position:
            return

        i = len(self.data) - 1
        today = self._dates[i]
        if today in self._traded_dates:
            return

        atr_v = float(self._atr_series[-1])
        if np.isnan(atr_v) or atr_v <= 0:
            return

        ah = self._asia_high[i]
        al = self._asia_low[i]
        if np.isnan(ah) or np.isnan(al):
            return

        rng = ah - al
        if not (0.5 * atr_v <= rng <= 2.0 * atr_v):
            return

        lstart = self._london_start[i]
        if lstart < 0:
            return
        bars_since_london = i - lstart
        if bars_since_london > 9:
            return

        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)
        if body < 1.2 * atr_v:
            return

        equity = float(self.equity)
        risk_pct = 0.5

        long_trigger = close > ah + 0.5 * atr_v
        short_trigger = close < al - 0.5 * atr_v

        if long_trigger:
            sl = close - 1.2 * atr_v
            tp = close + 2.5 * atr_v
            stop_dist = close - sl
            if stop_dist <= 0:
                return
            size = risk.lots_by_risk_pct(equity, risk_pct, stop_dist, close)
            if size <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            try:
                if isinstance(size, float) and 0 < size < 1:
                    self.buy(size=size, sl=sl, tp=tp)
                else:
                    self.buy(size=max(1, int(size)), sl=sl, tp=tp)
                self._traded_dates.add(today)
            except Exception:
                pass
        elif short_trigger:
            sl = close + 1.2 * atr_v
            tp = close - 2.5 * atr_v
            stop_dist = sl - close
            if stop_dist <= 0:
                return
            size = risk.lots_by_risk_pct(equity, risk_pct, stop_dist, close)
            if size <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            try:
                if isinstance(size, float) and 0 < size < 1:
                    self.sell(size=size, sl=sl, tp=tp)
                else:
                    self.sell(size=max(1, int(size)), sl=sl, tp=tp)
                self._traded_dates.add(today)
            except Exception:
                pass

    def _manage_open(self) -> None:
        if not self.position or not self.trades:
            return

        time_stop = 48
        trade = self.trades[-1]
        bars_open = len(self.data) - trade.entry_bar
        if bars_open >= time_stop:
            self.position.close()
            return

        atr_v = float(self._atr_series[-1])
        if np.isnan(atr_v) or atr_v <= 0:
            return

        price = float(self.data.Close[-1])
        for t in self.trades:
            entry = t.entry_price
            if t.is_long:
                initial_risk = entry - (entry - 1.2 * atr_v)
                r_move = price - entry
                if initial_risk > 0 and r_move >= 1.0 * initial_risk:
                    new_sl = price - 2.0 * atr_v
                    if t.sl is None or new_sl > t.sl:
                        t.sl = new_sl
            else:
                initial_risk = (entry + 1.2 * atr_v) - entry
                r_move = entry - price
                if initial_risk > 0 and r_move >= 1.0 * initial_risk:
                    new_sl = price + 2.0 * atr_v
                    if t.sl is None or new_sl < t.sl:
                        t.sl = new_sl

    def next(self):
        if not self._filters_ok():
            self._manage_open()
            return
        if not self._regime_ok():
            self._manage_open()
            return
        self._enter_if_signal()
        self._manage_open()
```
