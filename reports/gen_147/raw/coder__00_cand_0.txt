import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        super().init()
        self._bb_period = self.spec["entry_rule"]["params"]["bb_period"]
        self._bb_deviation = self.spec["entry_rule"]["params"]["bb_deviation"]
        self._rsi_period = self.spec["entry_rule"]["params"]["rsi_period"]
        self._rsi_thresholds = self.spec["entry_rule"]["params"]["rsi_thresholds"]
        self._atr_period = self.spec["stop_loss"]["params"]["atr_period"]
        self._atr_multiplier = self.spec["stop_loss"]["params"]["atr_multiplier"]
        self._tp_pips = self.spec["take_profit"]["params"]["pips"]
        self._time_stop_bars = self.spec["time_stop"]["params"]["bars"]
        self._lots = self.spec["sizing"]["params"]["lots"]
        self._bb_series = self.I(bollinger, self.data, self._bb_period, self._bb_deviation)
        self._rsi_series = self.I(rsi, self.data, self._rsi_period)
        self._atr_series = self.I(atr, self.data, self._atr_period)

    def _regime_ok(self):
        rf = self.spec["regime_filter"]
        if rf["type"] == "bb_width_percentile":
            bb_width_series = self.I(bb_width, self.data, rf["params"]["lookback"])
            percentile = rf["params"]["percentile"]
            threshold = np.percentile(bb_width_series, percentile)
            return bb_width_series[-1] < threshold
        return True

    def _filters_ok(self):
        return super()._filters_ok()

    def _enter_if_signal(self):
        if self._regime_ok() and self._filters_ok():
            upper_bb = self._bb_series[0][-1]
            lower_bb = self._bb_series[1][-1]
            close = self.data.Close[-1]
            rsi = self._rsi_series[-1]
            if (close >= upper_bb and rsi >= self._rsi_thresholds[1]) or (close <= lower_bb and rsi <= self._rsi_thresholds[0]):
                self.sl_price = close - self._atr_multiplier * self._atr_series[-1] if close > lower_bb else close + self._atr_multiplier * self._atr_series[-1]
                self.tp_price = close + self._tp_pips if close < upper_bb else close - self._tp_pips
                self.position.enter(self._lots)

    def _manage_open(self):
        if self.position:
            if self.spec["exit_rule"]["type"] == "opposite_bb":
                upper_bb = self._bb_series[0][-1]
                lower_bb = self._bb_series[1][-1]
                close = self.data.Close[-1]
                if (self.position.is_long and close <= lower_bb) or (not self.position.is_long and close >= upper_bb):
                    self.position.close()
            if self._time_stop_bars is not None:
                trade = self.trades[-1]
                bars_open = len(self.data) - trade.entry_bar
                if bars_open >= self._time_stop_bars:
                    self.position.close()
