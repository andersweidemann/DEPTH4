import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: Dict[str, Any] = {}
    _symbol: str = "XAUUSD"
    _equity_start: float = 10_000.0
    sl_price: Optional[float] = None
    tp_price: Optional[float] = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("filters", {}).get("session_utc") or []
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, sessions), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self.bb = self.I(bollinger, self.data, self.spec["regime_filter"]["params"]["bb_period"], self.spec["regime_filter"]["params"]["bb_deviation"])
        self.rsi = self.I(rsi, self.data, self.spec["entry_rule"]["params"]["rsi_period"])
        self.atr = self.I(atr, self.data, 14)

    def _regime_ok(self):
        bb_width_percentile = self.spec["regime_filter"]["params"]["percentile"]
        bb_width = bb_width(self.bb)
        bb_width_series = pd.Series(bb_width)
        percentile = np.percentile(bb_width_series, bb_width_percentile)
        return bb_width[-1] > percentile

    def _filters_ok(self):
        return super()._filters_ok()

    def _enter_if_signal(self):
        if self._regime_ok() and self._filters_ok():
            close = self.data.Close[-1]
            upper = self.bb[2][-1]
            lower = self.bb[0][-1]
            rsi = self.rsi[-1]
            if close > upper and rsi > self.spec["entry_rule"]["params"]["rsi_thresholds"][1]:
                self.position.enter(1, lots_by_risk_pct(self.spec["sizing_rule"]["params"]["fraction"], self._equity_start, self.data))
                self.sl_price = close - 1.5 * self.atr[-1]
                self.tp_price = lower
            elif close < lower and rsi < self.spec["entry_rule"]["params"]["rsi_thresholds"][0]:
                self.position.enter(-1, lots_by_risk_pct(self.spec["sizing_rule"]["params"]["fraction"], self._equity_start, self.data))
                self.sl_price = close + 1.5 * self.atr[-1]
                self.tp_price = upper

    def _manage_open(self):
        time_stop = self.spec["exit_rule"]["params"]["time_stop"]
        if self.position and time_stop is not None:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= time_stop:
                self.position.close()
        super()._manage_open()
