import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("regime_filter", {}).get("params", {}).get("start_hour", None) and self.spec.get("regime_filter", {}).get("params", {}).get("end_hour", None)
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, [(self.spec.get("regime_filter", {}).get("params", {}).get("start_hour", 0), self.spec.get("regime_filter", {}).get("params", {}).get("end_hour", 0))]), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self.asia_range_high = self.I(donchian, self.data, n=20, direction='high')
        self.asia_range_low = self.I(donchian, self.data, n=20, direction='low')
        self.atr = self.I(atr, self.data, n=20)

    def _regime_ok(self):
        return self._session_mask_full[-1] if self._session_mask_full is not None else True

    def _filters_ok(self):
        return self._regime_ok()

    def _enter_if_signal(self):
        if self._regime_ok():
            displacement = self.spec.get("entry_rule", {}).get("params", {}).get("displacement_atr", 1.0) * self.atr[-1]
            retest = self.spec.get("entry_rule", {}).get("params", {}).get("retest_atr", 0.0) * self.atr[-1]
            if self.data.Close[-1] > self.asia_range_high[-1] + displacement:
                self.position.open(long=True)
                self.sl_price = self.asia_range_low[-1] - retest
                self.tp_price = self.data.Close[-1] + self.spec.get("exit_rule", {}).get("params", {}).get("tp", 0.0)
            elif self.data.Close[-1] < self.asia_range_low[-1] - displacement:
                self.position.open(long=False)
                self.sl_price = self.asia_range_high[-1] + retest
                self.tp_price = self.data.Close[-1] - self.spec.get("exit_rule", {}).get("params", {}).get("tp", 0.0)

    def _manage_open(self):
        exit_cfg = self.spec.get("exit_rule", {})
        time_stop = exit_cfg.get("time_stop", 20)
        if not self.position:
            return
        if time_stop is not None:
            trade = self.trades[-1] if self.trades else None
            if trade is not None:
                bars_open = len(self.data) - trade.entry_bar
                if bars_open >= time_stop:
                    self.position.close()
                    return
        if self.tp_price is not None:
            if (self.position.is_long and self.data.Close[-1] >= self.tp_price) or (not self.position.is_long and self.data.Close[-1] <= self.tp_price):
                self.position.close()
                return
        if self.sl_price is not None:
            if (self.position.is_long and self.data.Close[-1] <= self.sl_price) or (not self.position.is_long and self.data.Close[-1] >= self.sl_price):
                self.position.close()
                return
