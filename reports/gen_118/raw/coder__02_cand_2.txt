import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.bollinger = self.I(bollinger, self.data, self.spec["entry_rule"]["params"]["bb_period"], self.spec["entry_rule"]["params"]["bb_deviation"])
        self.bb_width = self.I(bb_width, self.data, self.spec["entry_rule"]["params"]["bb_period"])
        self.atr = self.I(atr, self.data, 14)

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf["type"] == "bb_width":
            bb_width_val = float(self.bb_width[-1])
            if bb_width_val < rf["params"]["min_width"]:
                return False
        return True

    def _filters_ok(self):
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self):
        entry_cfg = self.spec.get("entry_rule")
        if entry_cfg["type"] == "reversal":
            bb_period = entry_cfg["params"]["bb_period"]
            bb_deviation = entry_cfg["params"]["bb_deviation"]
            reversal_threshold = entry_cfg["params"]["reversal_threshold"]
            bollinger = self.bollinger
            if len(bollinger) > 0:
                upper_bb = bollinger[-1][2]
                lower_bb = bollinger[-1][0]
                close = self.data.Close[-1]
                if close <= lower_bb and close / lower_bb - 1 <= -reversal_threshold / 100:
                    self.position.enter_long(lots_by_risk_pct(self.spec["sizing"]["params"]["size"], self.atr[-1]))
                    self.sl_price = close - self.spec["stop_loss"]["params"]["multiplier"] * self.atr[-1]
                    self.tp_price = bollinger[-1][1]
                elif close >= upper_bb and upper_bb / close - 1 <= -reversal_threshold / 100:
                    self.position.enter_short(lots_by_risk_pct(self.spec["sizing"]["params"]["size"], self.atr[-1]))
                    self.sl_price = close + self.spec["stop_loss"]["params"]["multiplier"] * self.atr[-1]
                    self.tp_price = bollinger[-1][1]

    def _manage_open(self):
        exit_cfg = self.spec.get("exit_rule")
        time_stop = self.spec.get("time_stop")
        if not self.position:
            return
        if time_stop["type"] == "bars" and time_stop["params"]["num_bars"] is not None:
            trade = self.trades[-1] if self.trades else None
            if trade is not None:
                bars_open = len(self.data) - trade.entry_bar
                if bars_open >= time_stop["params"]["num_bars"]:
                    self.position.close()
                    return
        if exit_cfg["type"] == "take_profit":
            if self.position.is_long and self.data.Close[-1] >= self.tp_price:
                self.position.close()
            elif not self.position.is_long and self.data.Close[-1] <= self.tp_price:
                self.position.close()
