import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: dict = {}
    _symbol: str = "GER40"
    _equity_start: float = 10_000.0
    sl_price: float = None
    tp_price: float = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("regime_filter", {}).get("params", {}).get("session_start") and self.spec.get("regime_filter", {}).get("params", {}).get("session_end")
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, [self.spec.get("regime_filter", {}).get("params", {}).get("session_start"), self.spec.get("regime_filter", {}).get("params", {}).get("session_end")]), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self.range_atr = self.I(atr, self.data, self.spec.get("entry_rule", {}).get("params", {}).get("range_atr_period"))

    def _regime_ok(self) -> bool:
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        session_start = rf.get("params", {}).get("session_start")
        session_end = rf.get("params", {}).get("session_end")
        current_time = pd.Timestamp(self.data.index[-1]).strftime("%H:%M")
        if session_start <= current_time < session_end:
            return True
        return False

    def _filters_ok(self) -> bool:
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self) -> None:
        entry_rule = self.spec.get("entry_rule", {})
        if entry_rule.get("type") == "breakout":
            range_atr = self.range_atr[-1]
            min_range_atr = entry_rule.get("params", {}).get("min_range_atr")
            max_range_atr = entry_rule.get("params", {}).get("max_range_atr")
            breakout_atr_multiplier = entry_rule.get("params", {}).get("breakout_atr_multiplier")
            if min_range_atr <= range_atr <= max_range_atr:
                if self.data.Close[-1] > self.data.High[-self.spec.get("entry_rule", {}).get("params", {}).get("range_atr_period")]:
                    self.sl_price = self.data.Close[-1] - breakout_atr_multiplier * range_atr
                    self.tp_price = self.data.Close[-1] + 500 * self.data._point
                    self.position.enter_long(lots_by_risk_pct(self.spec.get("sizing_rule", {}).get("params", {}).get("fraction"), self.equity, self.data))

    def _manage_open(self) -> None:
        exit_rule = self.spec.get("exit_rule", {})
        if exit_rule.get("type") == "multi_condition":
            conditions = exit_rule.get("params", {}).get("conditions")
            for condition in conditions:
                if condition.get("type") == "take_profit":
                    tp_price = condition.get("params", {}).get("tp_price")
                    if self.data.Close[-1] >= self.tp_price:
                        self.position.close()
                elif condition.get("type") == "stop_loss":
                    sl_price = condition.get("params", {}).get("sl_price")
                    if self.data.Close[-1] <= self.sl_price:
                        self.position.close()
                elif condition.get("type") == "time_stop":
                    time_stop_bar = condition.get("params", {}).get("time_stop_bar")
                    if len(self.data) - self.trades[-1].entry_bar >= time_stop_bar:
                        self.position.close()
