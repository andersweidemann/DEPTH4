import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: Dict[str, Any] = {}
    _symbol: str = "XAUUSD"
    _equity_start: float = 10_000.0
    sl_price: Optional[float] = None
    tp_price: Optional[float] = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("filters", {}).get("session_utc") or []
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, sessions), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self.bb_period = self.spec["regime_filter"]["params"]["period"]
        self.bb_deviation = self.spec["regime_filter"]["params"]["deviation"]
        self.bb_width_percentile = self.spec["regime_filter"]["params"]["percentile"]
        self.bb_touch_period = self.spec["entry_rule"]["params"]["bb_period"]
        self.bb_touch_deviation = self.spec["entry_rule"]["params"]["bb_deviation"]
        self.touch_tolerance = self.spec["entry_rule"]["params"]["touch_tolerance"]
        self.tp = self.spec["exit_rule"]["params"]["tp"]
        self.sl = self.spec["exit_rule"]["params"]["sl"]
        self.time_stop = self.spec["exit_rule"]["params"]["time_stop"]
        self.fraction = self.spec["sizing_rule"]["params"]["fraction"]
        self.I(signals.bollinger, self.data, self.bb_period, self.bb_deviation)
        self.I(signals.bb_width, self.data, self.bb_period, self.bb_deviation)
        self.I(signals.atr, self.data, 14)

    def _regime_ok(self):
        bb_width_val = float(self.I(signals.bb_width, self.data, self.bb_period, self.bb_deviation)[-1])
        bb_width_percentile_val = np.percentile(self.I(signals.bb_width, self.data, self.bb_period, self.bb_deviation), self.bb_width_percentile)
        if bb_width_val < bb_width_percentile_val:
            return True
        return False

    def _filters_ok(self):
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = self.spec.get("filters", {}).get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self):
        if self._regime_ok() and self._filters_ok():
            bb = self.I(signals.bollinger, self.data, self.bb_touch_period, self.bb_touch_deviation)
            close = self.data.Close
            if close[-1] <= bb[-1][0] * (1 + self.touch_tolerance) and close[-2] > bb[-2][0] * (1 + self.touch_tolerance):
                self.position.enter_long(lots_by_risk_pct(self.equity, self.fraction, self.data))
                self.sl_price = close[-1] - 1.5 * self.I(signals.atr, self.data, 14)[-1]
                self.tp_price = bb[-1][1]
            elif close[-1] >= bb[-1][1] * (1 - self.touch_tolerance) and close[-2] < bb[-2][1] * (1 - self.touch_tolerance):
                self.position.enter_short(lots_by_risk_pct(self.equity, self.fraction, self.data))
                self.sl_price = close[-1] + 1.5 * self.I(signals.atr, self.data, 14)[-1]
                self.tp_price = bb[-1][0]

    def _manage_open(self):
        exit_cfg = self.spec.get("exit_rule", {})
        time_stop = exit_cfg.get("time_stop")
        if not self.position:
            return
        trade = self.trades[-1]
        if time_stop is not None:
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= time_stop:
                self.position.close()
                return
        if self.tp == "opposite_bb":
            bb = self.I(signals.bollinger, self.data, self.bb_touch_period, self.bb_touch_deviation)
            if self.position.is_long and self.data.Close[-1] >= bb[-1][1]:
                self.position.close()
            elif not self.position.is_long and self.data.Close[-1] <= bb[-1][0]:
                self.position.close()
        if self.sl == "1.5_atr":
            atr = self.I(signals.atr, self.data, 14)
            if self.position.is_long and self.data.Close[-1] <= self.sl_price:
                self.position.close()
            elif not self.position.is_long and self.data.Close[-1] >= self.sl_price:
                self.position.close()
