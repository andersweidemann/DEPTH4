```python
import json
import os
from pathlib import Path
from typing import Any, Dict, Optional

import numpy as np
import pandas as pd

from agents import signals, regime, risk
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"

    donchian_period = 20
    adx_period = 14
    adx_min = 22
    ema_fast_p = 50
    ema_slow_p = 200
    atr_period = 14
    sl_atr_mult = 1.8
    tp_atr_mult = 3.0
    time_stop_bars = 48
    be_after_r = 1.0
    trail_atr_mult = 3.0
    trail_activate_r = 1.5
    risk_pct = 0.6
    atr_pct_min = 30.0
    sessions = [("08:00", "16:30")]

    def init(self):
        params = {}
        try:
            spec_file = Path(__file__).parent / self.spec_path
            if spec_file.exists():
                with open(spec_file) as f:
                    loaded = json.load(f)
                    if not self._spec:
                        self._spec = loaded
                    params = loaded.get("params", {})
        except Exception:
            pass

        self.spec = dict(self._spec) if self._spec else {}
        self._kill_state = risk.DailyKillState(start_of_day_equity=self._equity_start)

        p = self.spec.get("params", params) or params
        self.donchian_period = int(p.get("donchian_period", self.donchian_period))
        self.adx_period = int(p.get("adx_period", self.adx_period))
        self.adx_min = float(p.get("adx_min", self.adx_min))
        self.ema_fast_p = int(p.get("ema_fast", self.ema_fast_p))
        self.ema_slow_p = int(p.get("ema_slow", self.ema_slow_p))
        self.atr_period = int(p.get("atr_period", self.atr_period))
        self.sl_atr_mult = float(p.get("sl_atr_mult", self.sl_atr_mult))
        self.tp_atr_mult = float(p.get("tp_atr_mult", self.tp_atr_mult))

        rf = self.spec.get("regime_filter", {}) or {}
        rules = rf.get("rules", {}) if isinstance(rf, dict) else {}
        self.atr_pct_min = float(rules.get("atr_percentile_min", self.atr_pct_min))
        sess = rules.get("sessions_utc") or self.sessions
        self.sessions = [tuple(s) for s in sess]

        exit_cfg = self.spec.get("exit", {}) or {}
        self.time_stop_bars = int(exit_cfg.get("time_stop_bars", self.time_stop_bars)) \
            if exit_cfg.get("time_stop_bars") is not None else self.time_stop_bars

        self._atr_series = self.I(signals.atr, self.data, self.atr_period)
        self._ema_fast = self.I(signals.ema, self.data.Close, self.ema_fast_p)
        self._ema_slow = self.I(signals.ema, self.data.Close, self.ema_slow_p)
        self._adx_series = self.I(regime.adx, self.data, self.adx_period)

        dc = self.I(signals.donchian, self.data, self.donchian_period)
        self._dc_upper = dc[0]
        self._dc_lower = dc[1]

        self._atr_pct = self.I(regime.atr_percentile, self.data, self.atr_period, 200)

        full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        if self.sessions:
            self._session_mask_full = np.asarray(
                signals.session_mask(full_idx, self.sessions), dtype=bool)
        else:
            self._session_mask_full = None

        self._broker_spread_points = 0
        self._entry_price = None
        self._entry_atr = None
        self._is_long = None
        self._be_moved = False
        self._trail_active = False

    def _regime_ok(self) -> bool:
        adx_val = float(self._adx_series[-1]) if len(self._adx_series) else np.nan
        if np.isnan(adx_val) or adx_val < self.adx_min:
            return False
        atr_p = float(self._atr_pct[-1]) if len(self._atr_pct) else np.nan
        if not np.isnan(atr_p) and atr_p < self.atr_pct_min:
            return False
        return True

    def _filters_ok(self) -> bool:
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        idx = self.data.index
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        daily_kill_pct = (self.spec.get("sizing", {}) or {}).get("max_daily_loss_pct", 3.0)
        if not risk.daily_kill_ok(self._kill_state, now_date, self.equity, daily_kill_pct):
            return False
        return True

    def _enter_if_signal(self) -> None:
        if self.position:
            return
        if len(self.data) < max(self.ema_slow_p, self.donchian_period) + 2:
            return

        close = float(self.data.Close[-1])
        ema_f = float(self._ema_fast[-1])
        ema_s = float(self._ema_slow[-1])
        adx_v = float(self._adx_series[-1])
        atr_v = float(self._atr_series[-1])

        if np.isnan(ema_f) or np.isnan(ema_s) or np.isnan(adx_v) or np.isnan(atr_v):
            return
        if atr_v <= 0:
            return

        dc_hi_prior = float(self._dc_upper[-2])
        dc_lo_prior = float(self._dc_lower[-2])
        if np.isnan(dc_hi_prior) or np.isnan(dc_lo_prior):
            return

        long_sig = (close > dc_hi_prior) and (ema_f > ema_s) and (adx_v > self.adx_min)
        short_sig = (close < dc_lo_prior) and (ema_f < ema_s) and (adx_v > self.adx_min)

        if not (long_sig or short_sig):
            return

        if long_sig:
            sl = close - self.sl_atr_mult * atr_v
            tp = close + self.tp_atr_mult * atr_v
        else:
            sl = close + self.sl_atr_mult * atr_v
            tp = close - self.tp_atr_mult * atr_v

        stop_dist = abs(close - sl)
        if stop_dist <= 0:
            return

        size = risk.lots_by_risk_pct(
            equity=self.equity,
            risk_pct=self.risk_pct,
            stop_distance=stop_dist,
            price=close,
        )
        if size is None or size <= 0:
            return
        if isinstance(size, float) and size < 1:
            size = max(min(size, 0.999), 1e-4)
        else:
            size = max(int(size), 1)

        self.sl_price = sl
        self.tp_price = tp
        self._entry_price = close
        self._entry_atr = atr_v
        self._be_moved = False
        self._trail_active = False

        if long_sig:
            self._is_long = True
            self.buy(size=size, sl=sl, tp=tp)
        else:
            self._is_long = False
            self.sell(size=size, sl=sl, tp=tp)

    def _manage_open(self) -> None:
        if not self.position:
            self._entry_price = None
            self._entry_atr = None
            self._be_moved = False
            self._trail_active = False
            return

        if self.time_stop_bars and self.trades:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= self.time_stop_bars:
                self.position.close()
                return

        price = float(self.data.Close[-1])
        atr_now = float(self._atr_series[-1]) if len(self._atr_series) else np.nan
        if np.isnan(atr_now) or self._entry_price is None or self._entry_atr is None:
            return

        risk_dist = self.sl_atr_mult * self._entry_atr
        if risk_dist <= 0:
            return

        for trade in self.trades:
            if trade.is_long:
                r_mult = (price - trade.entry_price) / risk_dist
                if not self._be_moved and r_mult >= self.be_after_r:
                    be = trade.entry_price
                    if trade.sl is None or be > trade.sl:
                        trade.sl = be
                        self._be_moved = True
                if r_mult >= self.trail_activate_r:
                    new_sl = price - self.trail_atr_mult * atr_now
                    if trade.sl is None or new_sl > trade.sl:
                        trade.sl = new_sl
            else:
                r_mult = (trade.entry_price - price) / risk_dist
                if not self._be_moved and r_mult >= self.be_after_r:
                    be = trade.entry_price
                    if trade.sl is None or be < trade.sl:
                        trade.sl = be
                        self._be_moved = True
                if r_mult >= self.trail_activate_r:
                    new_sl = price + self.trail_atr_mult * atr_now
                    if trade.sl is None or new_sl < trade.sl:
                        trade.sl = new_sl

    def next(self):
        if not self._regime_ok():
            self._manage_open()
            return
        if not self._filters_ok():
            self._manage_open()
            return
        self._enter_if_signal()
        self._manage_open()
```
