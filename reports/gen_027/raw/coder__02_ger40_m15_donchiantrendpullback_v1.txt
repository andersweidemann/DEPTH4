```python
from __future__ import annotations

import json
import os
from typing import Any, Dict, Optional

import numpy as np
import pandas as pd

from agents import signals, regime, risk, config
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    # Strategy parameters
    _ema_fast = 20
    _ema_slow = 50
    _donchian_n = 40
    _breakout_lookback = 10
    _atr_n = 14
    _pullback_atr_mult = 0.3
    _sl_atr_mult = 2.0
    _tp_atr_mult = 3.5
    _time_stop_bars = 32
    _trail_atr_mult = 3.0
    _trail_activate_atr = 1.5
    _adx_min = 22.0
    _atr_pct_min = 35.0
    _risk_pct = 0.5
    _cooldown_bars = 6

    def init(self):
        try:
            here = os.path.dirname(os.path.abspath(__file__))
            spec_file = os.path.join(here, self.spec_path)
            if os.path.exists(spec_file):
                with open(spec_file, "r") as f:
                    loaded = json.load(f)
                    if isinstance(loaded, dict):
                        merged = dict(loaded)
                        if self._spec:
                            merged.update(self._spec)
                        type(self)._spec = merged
        except Exception:
            pass

        # Ensure session filter is present in spec for base class
        spec_dict = dict(self._spec) if self._spec else {}
        filters = dict(spec_dict.get("filters", {}))
        if "session_utc" not in filters:
            filters["session_utc"] = [["07:00", "16:00"]]
        spec_dict["filters"] = filters
        if "risk" not in spec_dict:
            spec_dict["risk"] = {}
        if "exit" not in spec_dict:
            spec_dict["exit"] = {
                "time_stop_bars": self._time_stop_bars,
                "trail_atr_mult": self._trail_atr_mult,
            }
        type(self)._spec = spec_dict

        super().init()

        self._ema_fast_series = self.I(signals.ema, self.data.Close, self._ema_fast)
        self._ema_slow_series = self.I(signals.ema, self.data.Close, self._ema_slow)
        self._atr_series = self.I(signals.atr, self.data, self._atr_n)

        dc = self.I(signals.donchian, self.data, self._donchian_n)
        if isinstance(dc, tuple) or (hasattr(dc, "ndim") and getattr(dc, "ndim", 1) > 1):
            self._dc_high = dc[0]
            self._dc_low = dc[1]
        else:
            self._dc_high = dc
            self._dc_low = self.I(
                lambda d, n: signals.donchian(d, n)[1] if isinstance(signals.donchian(d, n), tuple)
                else signals.donchian(d, n),
                self.data, self._donchian_n
            )

        self._adx_series = self.I(regime.adx, self.data, 14)
        self._atr_pct_series = self.I(regime.atr_percentile, self.data, 200)

        self._last_exit_bar = -10_000

    def _regime_ok(self) -> bool:
        adx_v = float(self._adx_series[-1]) if len(self._adx_series) else np.nan
        atr_p = float(self._atr_pct_series[-1]) if len(self._atr_pct_series) else np.nan
        if np.isnan(adx_v) or np.isnan(atr_p):
            return False
        if adx_v < self._adx_min:
            return False
        if atr_p < self._atr_pct_min:
            return False
        return True

    def _breakout_recent(self, direction: str) -> bool:
        lb = self._breakout_lookback
        n = len(self.data.Close)
        if n < self._donchian_n + lb + 2:
            return False
        highs = np.asarray(self.data.High)
        lows = np.asarray(self.data.Low)
        dch = np.asarray(self._dc_high)
        dcl = np.asarray(self._dc_low)
        start = max(0, n - lb)
        for i in range(start, n):
            if i - 1 < 0:
                continue
            prior_high = dch[i - 1]
            prior_low = dcl[i - 1]
            if direction == "long":
                if not np.isnan(prior_high) and highs[i] > prior_high:
                    return True
            else:
                if not np.isnan(prior_low) and lows[i] < prior_low:
                    return True
        return False

    def _enter_if_signal(self) -> None:
        if self.position:
            return
        bar_i = len(self.data) - 1
        if bar_i - self._last_exit_bar < self._cooldown_bars:
            return
        if not self._regime_ok():
            return
        if not self._filters_ok():
            return

        close = float(self.data.Close[-1])
        ema_f = float(self._ema_fast_series[-1])
        ema_s = float(self._ema_slow_series[-1])
        atr_v = float(self._atr_series[-1])
        if np.isnan(ema_f) or np.isnan(ema_s) or np.isnan(atr_v) or atr_v <= 0:
            return

        dist = abs(close - ema_f)
        pullback_ok = dist <= self._pullback_atr_mult * atr_v

        equity = float(self.equity)

        if ema_f > ema_s and close > ema_f and pullback_ok and self._breakout_recent("long"):
            sl = ema_f - self._sl_atr_mult * atr_v
            tp = close + self._tp_atr_mult * atr_v
            if sl >= close:
                return
            size = risk.lots_by_risk_pct(equity, self._risk_pct, close, sl, self._symbol)
            if size is None or size <= 0:
                return
            try:
                size_arg = int(size) if size >= 1 else float(size)
            except Exception:
                size_arg = size
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.buy(size=size_arg, sl=sl, tp=tp)
            except Exception:
                try:
                    self.buy(sl=sl, tp=tp)
                except Exception:
                    return

        elif ema_f < ema_s and close < ema_f and pullback_ok and self._breakout_recent("short"):
            sl = ema_f + self._sl_atr_mult * atr_v
            tp = close - self._tp_atr_mult * atr_v
            if sl <= close:
                return
            size = risk.lots_by_risk_pct(equity, self._risk_pct, close, sl, self._symbol)
            if size is None or size <= 0:
                return
            try:
                size_arg = int(size) if size >= 1 else float(size)
            except Exception:
                size_arg = size
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.sell(size=size_arg, sl=sl, tp=tp)
            except Exception:
                try:
                    self.sell(sl=sl, tp=tp)
                except Exception:
                    return

    def _manage_open(self) -> None:
        had_position = bool(self.position)

        if self.position and self.trades:
            atr_v = float(self._atr_series[-1]) if len(self._atr_series) else np.nan
            price = float(self.data.Close[-1])
            if not np.isnan(atr_v) and atr_v > 0:
                for trade in self.trades:
                    entry = float(trade.entry_price)
                    if trade.is_long:
                        if price - entry >= self._trail_activate_atr * atr_v:
                            new_sl = price - self._trail_atr_mult * atr_v
                            if trade.sl is None or new_sl > trade.sl:
                                trade.sl = new_sl
                    else:
                        if entry - price >= self._trail_activate_atr * atr_v:
                            new_sl = price + self._trail_atr_mult * atr_v
                            if trade.sl is None or new_sl < trade.sl:
                                trade.sl = new_sl

            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= self._time_stop_bars:
                self.position.close()

        if had_position and not self.position:
            self._last_exit_bar = len(self.data) - 1

    def next(self):
        if self.position:
            self._manage_open()
            if not self.position:
                self._last_exit_bar = len(self.data) - 1
        else:
            self._enter_if_signal()
```
