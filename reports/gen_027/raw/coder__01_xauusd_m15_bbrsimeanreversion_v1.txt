```python
from __future__ import annotations

import json
from pathlib import Path
from typing import Any, Dict, Optional

import numpy as np
import pandas as pd

from agents import signals, regime, risk, config
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"

    bb_period = 20
    bb_std = 1.75
    rsi_period = 7
    rsi_long_th = 10
    rsi_short_th = 90
    atr_period = 14
    adx_period = 14
    sl_atr_mult = 1.5
    time_stop_bars = 20
    cooldown_bars = 3
    risk_pct = 0.5
    bbw_lookback = 500
    bbw_min_pct = 30.0
    atrp_lookback = 200
    atrp_min = 25.0
    atrp_max = 85.0
    adx_max = 30.0
    session_start_hour = 7
    session_end_hour = 20

    def init(self):
        spec_file = Path(__file__).parent / self.spec_path
        if spec_file.exists():
            try:
                self._spec = json.loads(spec_file.read_text())
            except Exception:
                self._spec = {}
        super().init()

        close = self.data.Close

        def _mid(c, n, k):
            m, _, _ = signals.bollinger(c, n, k)
            return m

        def _up(c, n, k):
            _, u, _ = signals.bollinger(c, n, k)
            return u

        def _lo(c, n, k):
            _, _, l = signals.bollinger(c, n, k)
            return l

        self._bb_mid = self.I(_mid, close, self.bb_period, self.bb_std)
        self._bb_up = self.I(_up, close, self.bb_period, self.bb_std)
        self._bb_lo = self.I(_lo, close, self.bb_period, self.bb_std)

        self._bbw = self.I(signals.bb_width, close, self.bb_period, self.bb_std)
        self._rsi = self.I(signals.rsi, close, self.rsi_period)
        self._atr_series = self.I(signals.atr, self.data, self.atr_period)
        self._adx_series = self.I(regime.adx, self.data, self.adx_period)
        self._atr_pct_series = self.I(regime.atr_percentile, self.data,
                                       self.atr_period, self.atrp_lookback)

        def _bbw_pct(bbw_arr, lookback):
            arr = np.asarray(bbw_arr, dtype=float)
            out = np.full_like(arr, np.nan, dtype=float)
            for i in range(len(arr)):
                lo = max(0, i - lookback + 1)
                window = arr[lo:i + 1]
                window = window[~np.isnan(window)]
                if len(window) < 20:
                    continue
                cur = arr[i]
                if np.isnan(cur):
                    continue
                out[i] = 100.0 * (window <= cur).sum() / len(window)
            return out

        self._bbw_pct = self.I(_bbw_pct, self._bbw, self.bbw_lookback)

        self._last_exit_bar = -10_000

    def _regime_ok(self) -> bool:
        i = len(self.data) - 1
        if i < max(self.bb_period, self.adx_period, self.atr_period) + 5:
            return False

        bbw_p = float(self._bbw_pct[-1])
        if np.isnan(bbw_p) or bbw_p < self.bbw_min_pct:
            return False

        atrp = float(self._atr_pct_series[-1])
        if np.isnan(atrp) or atrp < self.atrp_min or atrp > self.atrp_max:
            return False

        adx_v = float(self._adx_series[-1])
        if np.isnan(adx_v) or adx_v >= self.adx_max:
            return False

        ts = pd.Timestamp(self.data.index[-1])
        hour = ts.hour
        if hour < self.session_start_hour or hour >= self.session_end_hour:
            return False

        return True

    def _filters_ok(self) -> bool:
        now_date = pd.Timestamp(self.data.index[-1]).strftime("%Y-%m-%d")
        try:
            dd_kill = self.spec.get("risk", {}).get(
                "daily_dd_kill_pct",
                config.load()["risk"]["daily_dd_kill_pct"])
        except Exception:
            dd_kill = 0.05
        if not risk.daily_kill_ok(self._kill_state, now_date, self.equity, dd_kill):
            return False
        return True

    def _enter_if_signal(self) -> None:
        if self.position:
            return

        bar_i = len(self.data) - 1
        if bar_i - self._last_exit_bar < self.cooldown_bars:
            return

        close = float(self.data.Close[-1])
        upper = float(self._bb_up[-1])
        lower = float(self._bb_lo[-1])
        mid = float(self._bb_mid[-1])
        r = float(self._rsi[-1])
        atr_v = float(self._atr_series[-1])

        if any(np.isnan(x) for x in (upper, lower, mid, r, atr_v)) or atr_v <= 0:
            return

        long_sig = close < lower and r < self.rsi_long_th
        short_sig = close > upper and r > self.rsi_short_th

        if not (long_sig or short_sig):
            return

        if long_sig:
            sl = close - self.sl_atr_mult * atr_v
            tp = upper
            if sl >= close or tp <= close:
                return
            risk_per_unit = close - sl
            size = risk.lots_by_risk_pct(self.equity, self.risk_pct / 100.0,
                                         risk_per_unit, self._symbol)
            size = self._normalize_size(size)
            if size <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            self.buy(size=size, sl=sl, tp=tp)
        else:
            sl = close + self.sl_atr_mult * atr_v
            tp = lower
            if sl <= close or tp >= close:
                return
            risk_per_unit = sl - close
            size = risk.lots_by_risk_pct(self.equity, self.risk_pct / 100.0,
                                         risk_per_unit, self._symbol)
            size = self._normalize_size(size)
            if size <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            self.sell(size=size, sl=sl, tp=tp)

    def _normalize_size(self, size: float) -> float:
        if size is None or not np.isfinite(size) or size <= 0:
            return 0.0
        if size >= 1:
            return int(size)
        return max(0.0, min(0.999, float(size)))

    def _manage_open(self) -> None:
        if not self.position:
            return

        trade = self.trades[-1] if self.trades else None
        if trade is None:
            return

        bars_open = len(self.data) - trade.entry_bar

        if trade.is_long:
            mid = float(self._bb_mid[-1])
            if bars_open >= 3 and not np.isnan(mid):
                if float(self.data.High[-1]) >= mid and trade.tp is not None and mid < trade.tp:
                    trade.tp = mid
        else:
            mid = float(self._bb_mid[-1])
            if bars_open >= 3 and not np.isnan(mid):
                if float(self.data.Low[-1]) <= mid and trade.tp is not None and mid > trade.tp:
                    trade.tp = mid

        if bars_open >= self.time_stop_bars:
            self._last_exit_bar = len(self.data) - 1
            self.position.close()
            return

    def next(self):
        if not self.position and self.trades == [] and hasattr(self, "_prev_had_position"):
            if self._prev_had_position:
                self._last_exit_bar = len(self.data) - 1
        self._prev_had_position = bool(self.position)

        if not self._filters_ok():
            self._manage_open()
            return
        if not self._regime_ok():
            self._manage_open()
            return
        self._manage_open()
        self._enter_if_signal()
```
