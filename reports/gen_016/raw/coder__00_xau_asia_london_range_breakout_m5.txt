```python
import json
import os
from pathlib import Path
from typing import Any, Dict, Optional

import numpy as np
import pandas as pd

from agents import signals, regime, risk, config
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    def init(self):
        super().init()
        spec_file = Path(__file__).parent / self.spec_path
        if spec_file.exists():
            try:
                with open(spec_file, "r") as f:
                    loaded = json.load(f)
                if isinstance(loaded, dict):
                    self.spec.update(loaded)
            except Exception:
                pass

        self._atr_series = self.I(signals.atr, self.data, 14)
        self._adx_series = self.I(regime.adx, self.data, 14)
        self._atr_pct_series = self.I(regime.atr_percentile, self.data, 14, 200)

        dc = self.I(signals.donchian, self.data, 72)
        if isinstance(dc, tuple) or (hasattr(dc, "ndim") and getattr(dc, "ndim", 1) == 2):
            self._dc_upper = dc[0]
            self._dc_lower = dc[1]
        else:
            self._dc_upper = dc
            self._dc_lower = dc

        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        self._asia_mask = np.asarray(signals.session_mask(idx, ["00:00-06:00"]), dtype=bool)
        self._london_mask = np.asarray(signals.session_mask(idx, ["07:00-10:00"]), dtype=bool)

        self._asia_high = np.full(len(idx), np.nan)
        self._asia_low = np.full(len(idx), np.nan)
        current_high = -np.inf
        current_low = np.inf
        last_date = None
        highs = np.asarray(self.data.High)
        lows = np.asarray(self.data.Low)
        for i in range(len(idx)):
            ts = pd.Timestamp(idx[i])
            d = ts.date()
            if d != last_date:
                current_high = -np.inf
                current_low = np.inf
                last_date = d
            if self._asia_mask[i]:
                if highs[i] > current_high:
                    current_high = highs[i]
                if lows[i] < current_low:
                    current_low = lows[i]
            if current_high != -np.inf:
                self._asia_high[i] = current_high
                self._asia_low[i] = current_low

        self._last_trade_date: Optional[Any] = None
        self._day_entered: Optional[Any] = None

    def _regime_ok(self) -> bool:
        if len(self._adx_series) == 0:
            return False
        adx_val = float(self._adx_series[-1])
        atr_pct = float(self._atr_pct_series[-1])
        if np.isnan(adx_val) or np.isnan(atr_pct):
            return False
        if adx_val < 18:
            return False
        if atr_pct < 35:
            return False
        return True

    def _filters_ok(self) -> bool:
        bar_i = len(self.data) - 1
        if bar_i < 0 or bar_i >= len(self._london_mask):
            return False
        if not bool(self._london_mask[bar_i]):
            return False
        idx = self.data.index
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        try:
            dd_kill = self.spec.get("risk", {}).get(
                "daily_dd_kill_pct",
                config.load()["risk"]["daily_dd_kill_pct"],
            )
        except Exception:
            dd_kill = 2.0
        if not risk.daily_kill_ok(self._kill_state, now_date, self.equity, dd_kill):
            return False
        return True

    def _enter_if_signal(self) -> None:
        if self.position:
            return
        bar_i = len(self.data) - 1
        ts = pd.Timestamp(self.data.index[-1])
        today = ts.date()

        if self._day_entered == today:
            return

        atr_now = float(self._atr_series[-1])
        if np.isnan(atr_now) or atr_now <= 0:
            return

        asia_hi = float(self._asia_high[bar_i])
        asia_lo = float(self._asia_low[bar_i])
        if np.isnan(asia_hi) or np.isnan(asia_lo):
            return

        asia_range = asia_hi - asia_lo
        if asia_range < 0.5 * atr_now or asia_range > 2.0 * atr_now:
            return

        close = float(self.data.Close[-1])
        open_ = float(self.data.Open[-1])
        body = abs(close - open_)
        if body < 1.2 * atr_now:
            return

        long_level = asia_hi + 0.5 * atr_now
        short_level = asia_lo - 0.5 * atr_now

        go_long = close > long_level
        go_short = close < short_level

        if not (go_long or go_short):
            return

        if go_long:
            entry = close
            sl = long_level - 0.75 * atr_now
            tp = entry + 2.0 * atr_now
            risk_per_unit = entry - sl
            if risk_per_unit <= 0:
                return
            try:
                size = risk.lots_by_risk_pct(
                    equity=self.equity,
                    risk_pct=0.5,
                    stop_distance=risk_per_unit,
                    symbol=self._symbol,
                )
            except Exception:
                size = max(1, int((self.equity * 0.005) / risk_per_unit))
            size = max(1, int(size)) if not isinstance(size, float) or size >= 1 else size
            if isinstance(size, float) and size <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.buy(size=size, sl=sl, tp=tp)
            except Exception:
                try:
                    self.buy(sl=sl, tp=tp)
                except Exception:
                    return
            self._day_entered = today
        elif go_short:
            entry = close
            sl = short_level + 0.75 * atr_now
            tp = entry - 2.0 * atr_now
            risk_per_unit = sl - entry
            if risk_per_unit <= 0:
                return
            try:
                size = risk.lots_by_risk_pct(
                    equity=self.equity,
                    risk_pct=0.5,
                    stop_distance=risk_per_unit,
                    symbol=self._symbol,
                )
            except Exception:
                size = max(1, int((self.equity * 0.005) / risk_per_unit))
            size = max(1, int(size)) if not isinstance(size, float) or size >= 1 else size
            if isinstance(size, float) and size <= 0:
                return
            self.sl_price = sl
            self.tp_price = tp
            try:
                self.sell(size=size, sl=sl, tp=tp)
            except Exception:
                try:
                    self.sell(sl=sl, tp=tp)
                except Exception:
                    return
            self._day_entered = today

    def _manage_open(self) -> None:
        if not self.position:
            return
        time_stop = 36
        if self.trades:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= time_stop:
                self.position.close()
                return

        atr_now = float(self._atr_series[-1])
        if np.isnan(atr_now) or atr_now <= 0:
            return
        price = float(self.data.Close[-1])
        for trade in self.trades:
            entry = trade.entry_price
            if trade.is_long:
                if price - entry >= 1.0 * atr_now:
                    if trade.sl is None or trade.sl < entry:
                        trade.sl = entry
            else:
                if entry - price >= 1.0 * atr_now:
                    if trade.sl is None or trade.sl > entry:
                        trade.sl = entry

    def next(self):
        if not self._regime_ok():
            self._manage_open()
            return
        if not self._filters_ok():
            self._manage_open()
            return
        self._enter_if_signal()
        self._manage_open()
```
