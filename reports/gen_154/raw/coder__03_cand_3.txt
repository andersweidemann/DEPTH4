import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self._atr_series = self.I(atr, self.data, self.spec["regime_filter"]["params"]["period"])
        self._donchian_series = self.I(donchian, self.data, self.spec["entry_rule"]["params"]["breakout_period"])
        self._session_mask_full = None

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        atr_percentile_val = float(self.I(atr_percentile, self.data, rf["params"]["period"], rf["params"]["percentile"])[-1])
        return atr_percentile_val > 0

    def _filters_ok(self):
        return True

    def _enter_if_signal(self):
        if self._regime_ok() and self._filters_ok():
            breakout_price = float(self._donchian_series[-1])
            if self.data.Close[-1] > breakout_price:
                size = lots_by_risk_pct(self._spec, self.data, self._atr_series[-1])
                self.position.enter(long=True, size=size)
                self.sl_price = self.data.Close[-1] - self._atr_series[-1]
                self.tp_price = self.data.Close[-1] + self._atr_series[-1]

    def _manage_open(self):
        time_stop = self.spec["exit_rule"]["params"]["time_stop_bars"]
        if self.position and time_stop is not None:
            bars_open = len(self.data) - self.position.entry_bar
            if bars_open >= time_stop:
                self.position.close()
