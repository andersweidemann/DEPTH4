import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: Dict[str, Any] = {}
    _symbol: str = "BTCUSD"
    _equity_start: float = 10_000.0
    sl_price: Optional[float] = None
    tp_price: Optional[float] = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.donchian_channel_high = self.I(signals.donchian, self.data, n=20, high_low='high')
        self.donchian_channel_low = self.I(signals.donchian, self.data, n=20, high_low='low')
        self.atr = self.I(signals.atr, self.data, n=14)
        self.atr_percentile = self.I(adx.atr_percentile, self.data, n=14, percentile=50)

    def _regime_ok(self) -> bool:
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf.get("type") == "atr_percentile":
            atr_now = float(self.atr[-1])
            atr_percentile_now = float(self.atr_percentile[-1])
            return atr_now > atr_percentile_now
        return True

    def _filters_ok(self) -> bool:
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self) -> None:
        entry_rules = self.spec.get("entry_rules", {})
        long_condition = entry_rules.get("long", {}).get("condition")
        short_condition = entry_rules.get("short", {}).get("condition")
        if long_condition and self.position.is_empty:
            high = float(self.data.High[-1])
            donchian_channel_high_now = float(self.donchian_channel_high[-1])
            atr_now = float(self.atr[-1])
            atr_percentile_now = float(self.atr_percentile[-1])
            if high > donchian_channel_high_now and atr_now > atr_percentile_now:
                self.position.enter(long=True, size=lots_by_risk_pct(self.spec, self.data, self.equity))
                self.sl_price = float(self.data.Low[-1]) - 100 * self.data.pip
                self.tp_price = float(self.data.High[-1]) + 500 * self.data.pip
        elif short_condition and self.position.is_empty:
            low = float(self.data.Low[-1])
            donchian_channel_low_now = float(self.donchian_channel_low[-1])
            atr_now = float(self.atr[-1])
            atr_percentile_now = float(self.atr_percentile[-1])
            if low < donchian_channel_low_now and atr_now > atr_percentile_now:
                self.position.enter(long=False, size=lots_by_risk_pct(self.spec, self.data, self.equity))
                self.sl_price = float(self.data.High[-1]) + 100 * self.data.pip
                self.tp_price = float(self.data.Low[-1]) - 500 * self.data.pip

    def _manage_open(self) -> None:
        exit_cfg = self.spec.get("exit_rules", {})
        time_stop = exit_cfg.get("time_stop", {}).get("hours")
        if not self.position:
            return
        if time_stop is not None:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= time_stop * 12:
                self.position.close()
                return
        sl = exit_cfg.get("sl", {}).get("pips")
        tp = exit_cfg.get("tp", {}).get("pips")
        if sl is not None and tp is not None:
            if self.position.is_long and self.data.Low[-1] <= self.sl_price:
                self.position.close()
            elif not self.position.is_long and self.data.High[-1] >= self.sl_price:
                self.position.close()
            if self.position.is_long and self.data.High[-1] >= self.tp_price:
                self.position.close()
            elif not self.position.is_long and self.data.Low[-1] <= self.tp_price:
                self.position.close()
