import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        sessions = self.spec.get("filters", {}).get("session_utc") or []
        if sessions:
            full_idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
            self._session_mask_full = np.asarray(session_mask(full_idx, sessions), dtype=bool)
        else:
            self._session_mask_full = None
        self._broker_spread_points = 0
        self.bb_period = self.spec["entry_rule"]["params"]["bb_period"]
        self.bb_deviation = self.spec["entry_rule"]["params"]["bb_deviation"]
        self.rejection_threshold = self.spec["entry_rule"]["params"]["rejection_threshold"]
        self.time_stop = self.spec["exit_rule"]["time_stop"]
        self.sl_mult = float(self.spec["exit_rule"]["sl"].split("*")[0])
        self.tp_price = None
        self.sl_price = None
        self.I(signals.bollinger, self.data, self.bb_period, self.bb_deviation)
        self.I(signals.atr, self.data, 14)

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf["type"] == "bb_width_percentile":
            bb_width_series = self.I(signals.bb_width, self.data, self.bb_period, self.bb_deviation)
            threshold = rf["threshold"]
            bb_width_now = float(bb_width_series[-1])
            bb_width_percentile = np.percentile(bb_width_series, threshold)
            return bb_width_now < bb_width_percentile
        return True

    def _filters_ok(self):
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self):
        bb = self.I(signals.bollinger, self.data, self.bb_period, self.bb_deviation)
        close = self.data.Close
        upper_bb = bb.upper
        lower_bb = bb.lower
        if close[-1] > upper_bb[-1] * (1 + self.rejection_threshold):
            self.position.enter_short(lots_by_risk_pct(self.spec["sizing_rule"]["fraction"], self.equity, self.data))
            self.sl_price = upper_bb[-1] + self.sl_mult * self.I(signals.atr, self.data, 14)[-1]
            self.tp_price = bb.middle[-1]
        elif close[-1] < lower_bb[-1] * (1 - self.rejection_threshold):
            self.position.enter_long(lots_by_risk_pct(self.spec["sizing_rule"]["fraction"], self.equity, self.data))
            self.sl_price = lower_bb[-1] - self.sl_mult * self.I(signals.atr, self.data, 14)[-1]
            self.tp_price = bb.middle[-1]

    def _manage_open(self):
        exit_cfg = self.spec.get("exit_rule", {})
        time_stop = exit_cfg.get("time_stop")
        if not self.position:
            return
        trade = self.trades[-1]
        bars_open = len(self.data) - trade.entry_bar
        if time_stop is not None and bars_open >= time_stop:
            self.position.close()
            return
        bb = self.I(signals.bollinger, self.data, self.bb_period, self.bb_deviation)
        if self.position.is_long:
            self.sl_price = min(self.sl_price, bb.lower[-1] - self.sl_mult * self.I(signals.atr, self.data, 14)[-1])
        else:
            self.sl_price = max(self.sl_price, bb.upper[-1] + self.sl_mult * self.I(signals.atr, self.data, 14)[-1])
