import json
import os
from pathlib import Path
from typing import Optional

import numpy as np
import pandas as pd

from agents import signals, regime, risk
from agents.backtester import RegimeStrategy


class Strategy(RegimeStrategy):
    spec_path = "spec.json"

    atr_period = 14
    adx_period = 14
    atr_pct_lookback = 200

    asia_start_hour = 0
    asia_end_hour = 6
    london_start_hour = 7
    london_end_hour = 10

    range_atr_min = 0.5
    range_atr_max = 2.0
    breakout_atr_mult = 0.5
    body_atr_mult = 1.2
    max_bars_since_open = 9

    sl_atr_mult = 0.75
    tp_atr_mult = 2.0
    time_stop_bars = 36
    breakeven_atr_mult = 1.0
    trail_atr_mult = 2.0

    risk_per_trade_pct = 0.5
    max_daily_trades = 2
    adx_min = 18
    atr_pct_min = 30
    atr_pct_max = 95

    def init(self):
        super().init()

        self._atr_series = self.I(signals.atr, self.data, self.atr_period)
        self._adx_series = self.I(regime.adx, self.data, self.adx_period)
        self._atr_pct_series = self.I(
            regime.atr_percentile, self.data, self.atr_period, self.atr_pct_lookback
        )

        idx = self.data.df.index if hasattr(self.data, "df") else self.data.index
        ts = pd.DatetimeIndex(idx)
        if ts.tz is not None:
            ts = ts.tz_convert("UTC").tz_localize(None)

        hours = ts.hour.values
        minutes = ts.minute.values
        dates = ts.normalize().values

        asia_mask = (hours >= self.asia_start_hour) & (hours < self.asia_end_hour)
        london_mask = (hours >= self.london_start_hour) & (hours < self.london_end_hour)

        self._asia_mask = asia_mask
        self._london_mask = london_mask
        self._bar_dates = dates
        self._hours = hours
        self._minutes = minutes

        n = len(ts)
        high = np.asarray(self.data.High)
        low = np.asarray(self.data.Low)

        asia_high = np.full(n, np.nan)
        asia_low = np.full(n, np.nan)

        cur_date = None
        cur_hi = -np.inf
        cur_lo = np.inf
        finalized_hi = np.nan
        finalized_lo = np.nan

        for i in range(n):
            d = dates[i]
            if cur_date is None or d != cur_date:
                cur_date = d
                cur_hi = -np.inf
                cur_lo = np.inf
                finalized_hi = np.nan
                finalized_lo = np.nan

            if asia_mask[i]:
                cur_hi = max(cur_hi, high[i])
                cur_lo = min(cur_lo, low[i])
                finalized_hi = cur_hi if cur_hi != -np.inf else np.nan
                finalized_lo = cur_lo if cur_lo != np.inf else np.nan
            elif hours[i] >= self.asia_end_hour:
                if np.isnan(finalized_hi) and cur_hi != -np.inf:
                    finalized_hi = cur_hi
                    finalized_lo = cur_lo

            asia_high[i] = finalized_hi
            asia_low[i] = finalized_lo

        self._asia_high = asia_high
        self._asia_low = asia_low

        london_open_bar = np.full(n, -1, dtype=int)
        cur_date = None
        open_bar = -1
        for i in range(n):
            d = dates[i]
            if cur_date is None or d != cur_date:
                cur_date = d
                open_bar = -1
            if london_mask[i] and open_bar < 0:
                open_bar = i
            london_open_bar[i] = open_bar

        self._london_open_bar = london_open_bar

        self._trade_day = None
        self._trades_today = 0
        self._breakout_done_day = None
        self._breakout_done = False
        self._be_moved = {}

    def _regime_ok(self) -> bool:
        i = len(self.data) - 1
        if i < 0:
            return False

        adx_val = float(self._adx_series[-1])
        if np.isnan(adx_val) or adx_val < self.adx_min:
            return False

        atr_pct = float(self._atr_pct_series[-1])
        if np.isnan(atr_pct):
            return False
        if atr_pct < self.atr_pct_min or atr_pct > self.atr_pct_max:
            return False

        if not self._london_mask[i]:
            return False

        return True

    def _filters_ok(self) -> bool:
        i = len(self.data) - 1
        today = self._bar_dates[i]
        if self._trade_day != today:
            self._trade_day = today
            self._trades_today = 0
        if self._trades_today >= self.max_daily_trades:
            return False
        return True

    def _enter_if_signal(self) -> None:
        if self.position:
            return

        i = len(self.data) - 1
        if i < 1:
            return

        today = self._bar_dates[i]
        if self._breakout_done_day != today:
            self._breakout_done_day = today
            self._breakout_done = False

        if self._breakout_done:
            return

        atr_val = float(self._atr_series[-1])
        if np.isnan(atr_val) or atr_val <= 0:
            return

        a_hi = self._asia_high[i]
        a_lo = self._asia_low[i]
        if np.isnan(a_hi) or np.isnan(a_lo):
            return

        rng = a_hi - a_lo
        if rng < self.range_atr_min * atr_val or rng > self.range_atr_max * atr_val:
            return

        open_bar = self._london_open_bar[i]
        if open_bar < 0:
            return
        bars_since_open = i - open_bar
        if bars_since_open > self.max_bars_since_open:
            return

        close = float(self.data.Close[-1])
        open_p = float(self.data.Open[-1])
        body = abs(close - open_p)
        if body < self.body_atr_mult * atr_val:
            return

        long_trigger = a_hi + self.breakout_atr_mult * atr_val
        short_trigger = a_lo - self.breakout_atr_mult * atr_val

        direction = 0
        if close > long_trigger:
            direction = 1
        elif close < short_trigger:
            direction = -1
        else:
            return

        if direction > 0:
            sl = close - self.sl_atr_mult * atr_val
            tp = close + self.tp_atr_mult * atr_val
        else:
            sl = close + self.sl_atr_mult * atr_val
            tp = close - self.tp_atr_mult * atr_val

        stop_dist = abs(close - sl)
        if stop_dist <= 0:
            return

        equity = float(self.equity)
        try:
            size = risk.lots_by_risk_pct(
                equity=equity,
                risk_pct=self.risk_per_trade_pct,
                stop_distance=stop_dist,
                price=close,
                symbol=self._symbol,
            )
        except TypeError:
            size = risk.lots_by_risk_pct(
                equity, self.risk_per_trade_pct, stop_dist, close
            )

        if size is None or size <= 0:
            risk_amount = equity * (self.risk_per_trade_pct / 100.0)
            units = risk_amount / stop_dist
            size = max(1, int(units))

        if isinstance(size, float):
            if size <= 0:
                return
            if size >= 1:
                size = int(size)
            else:
                if not (0 < size < 1):
                    return

        self.sl_price = sl
        self.tp_price = tp

        try:
            if direction > 0:
                self.buy(size=size, sl=sl, tp=tp)
            else:
                self.sell(size=size, sl=sl, tp=tp)
            self._trades_today += 1
            self._breakout_done = True
        except Exception:
            return

    def _manage_open(self) -> None:
        if not self.position or not self.trades:
            return

        atr_val = float(self._atr_series[-1])
        price = float(self.data.Close[-1])

        for trade in self.trades:
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= self.time_stop_bars:
                trade.close()
                continue

            if np.isnan(atr_val) or atr_val <= 0:
                continue

            tid = trade.entry_bar
            moved = self._be_moved.get(tid, False)

            if trade.is_long:
                if not moved and price - trade.entry_price >= self.breakeven_atr_mult * atr_val:
                    if trade.sl is None or trade.sl < trade.entry_price:
                        trade.sl = trade.entry_price
                    self._be_moved[tid] = True
                if trade.pl_pct > 0:
                    new_sl = price - self.trail_atr_mult * atr_val
                    if trade.sl is None or new_sl > trade.sl:
                        trade.sl = new_sl
            else:
                if not moved and trade.entry_price - price >= self.breakeven_atr_mult * atr_val:
                    if trade.sl is None or trade.sl > trade.entry_price:
                        trade.sl = trade.entry_price
                    self._be_moved[tid] = True
                if trade.pl_pct > 0:
                    new_sl = price + self.trail_atr_mult * atr_val
                    if trade.sl is None or new_sl < trade.sl:
                        trade.sl = new_sl

    def next(self):
        if not self._regime_ok():
            self._manage_open()
            return
        if not self._filters_ok():
            self._manage_open()
            return
        self._enter_if_signal()
        self._manage_open()
