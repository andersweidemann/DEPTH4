import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: dict = {}
    _symbol: str = "BTCUSD"
    _equity_start: float = 10_000.0
    sl_price: float = None
    tp_price: float = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.bollinger_bands = self.I(bollinger, self.data, n=self.spec["regime_filter"]["params"]["bb_period"], dev=self.spec["regime_filter"]["params"]["bb_deviation"])
        self.bb_width = self.I(bb_width, self.data, n=self.spec["regime_filter"]["params"]["bb_period"], dev=self.spec["regime_filter"]["params"]["bb_deviation"])
        self.atr = self.I(atr, self.data, n=self.spec["entry_rule"]["params"]["atr_period"])
        self._session_mask_full = None

    def _regime_ok(self) -> bool:
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf["type"] == "bb_width_decrease":
            bb_width_decrease = self.bb_width[-1] - self.bb_width[-2]
            return bb_width_decrease < self.spec["regime_filter"]["params"]["decrease_threshold"]
        return True

    def _filters_ok(self) -> bool:
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self) -> None:
        er = self.spec.get("entry_rule")
        if er["type"] == "volatility_contraction":
            atr_now = float(self.atr[-1])
            atr_prev = float(self.atr[-2])
            if atr_now < atr_prev:
                self.sl_price = self.data.Close[-1] - 2 * atr_now
                self.tp_price = self.data.Close[-1] + 2 * atr_now
                lots = lots_by_risk_pct(self.spec["sizing_rule"]["params"]["fraction"], self.equity, self.data.Close[-1], self.sl_price)
                self.position.enter(lots)

    def _manage_open(self) -> None:
        exit_cfg = self.spec.get("exit_rule")
        time_stop = exit_cfg.get("time_stop")
        if not self.position:
            return
        if time_stop is not None:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= time_stop:
                self.position.close()
                return
        if exit_cfg["params"]["tp"] == "opposite_bb":
            upper_bb = self.bollinger_bands[-1][1]
            lower_bb = self.bollinger_bands[-1][0]
            if self.position.is_long and self.data.Close[-1] >= upper_bb:
                self.position.close()
            elif not self.position.is_long and self.data.Close[-1] <= lower_bb:
                self.position.close()
        if exit_cfg["params"]["sl"] == "2_atr":
            atr_now = float(self.atr[-1])
            if self.position.is_long and self.data.Close[-1] <= self.position.entry_price - 2 * atr_now:
                self.position.close()
            elif not self.position.is_long and self.data.Close[-1] >= self.position.entry_price + 2 * atr_now:
                self.position.close()
