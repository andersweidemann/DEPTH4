import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    spec_path: str = "spec.json"
    _spec: Dict[str, Any] = {}
    _symbol: str = "XAUUSD"
    _equity_start: float = 10_000.0
    sl_price: Optional[float] = None
    tp_price: Optional[float] = None

    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self._atr_series = self.I(atr, self.data, 14)
        self._rsi_series = self.I(rsi, self.data, 14)
        self._breakout_series = self.I(atr_breakout_levels, self.data, 20)
        self._upper_range = self._breakout_series['upper']
        self._lower_range = self._breakout_series['lower']

    def _regime_ok(self) -> bool:
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf.get("type") == "atr_percentile":
            percentile = rf.get("params", {}).get("percentile")
            atr_percentile_series = self.I(atr_percentile, self.data, 14, percentile)
            return atr_percentile_series[-1] > 0
        return True

    def _filters_ok(self) -> bool:
        filters = self.spec.get("filters", {})
        idx = self.data.index
        bar_i = len(self.data) - 1
        mask = getattr(self, "_session_mask_full", None)
        if mask is not None and 0 <= bar_i < len(mask):
            if not bool(mask[bar_i]):
                return False
        max_spread = filters.get("max_spread_points")
        if max_spread is not None:
            broker_spread = self._broker_spread_points
            if not spread_ok(broker_spread, max_spread):
                return False
        now_date = pd.Timestamp(idx[-1]).strftime("%Y-%m-%d")
        if not daily_kill_ok(self._kill_state, now_date, self.equity, self.spec.get("risk", {}).get("daily_dd_kill_pct", 0.2)):
            return False
        return True

    def _enter_if_signal(self) -> None:
        entry_rules = self.spec.get("entry_rules")
        if entry_rules:
            long_condition = entry_rules.get("long", {}).get("condition")
            short_condition = entry_rules.get("short", {}).get("condition")
            if long_condition and short_condition:
                if long_condition == "breakout > upper_range && rsi < 50":
                    if self._breakout_series['breakout'][-1] > self._upper_range[-1] and self._rsi_series[-1] < 50:
                        self.position.open_long()
                        self.sl_price = self.data.Close[-1] - 150 * self.data._pip
                        self.tp_price = self.data.Close[-1] + 600 * self.data._pip
                if short_condition == "breakout < lower_range && rsi > 50":
                    if self._breakout_series['breakout'][-1] < self._lower_range[-1] and self._rsi_series[-1] > 50:
                        self.position.open_short()
                        self.sl_price = self.data.Close[-1] + 150 * self.data._pip
                        self.tp_price = self.data.Close[-1] - 600 * self.data._pip

    def _manage_open(self) -> None:
        exit_cfg = self.spec.get("exit_rules")
        time_stop = exit_cfg.get("time_stop", {}).get("params", {}).get("num_bars")
        if not self.position:
            return
        if time_stop is not None:
            trade = self.trades[-1]
            bars_open = len(self.data) - trade.entry_bar
            if bars_open >= time_stop:
                self.position.close()
                return
        sl = exit_cfg.get("sl", {}).get("params", {}).get("num_pips")
        tp = exit_cfg.get("tp", {}).get("params", {}).get("num_pips")
        if sl is not None and tp is not None:
            if self.position.is_long:
                if self.data.Close[-1] < self.sl_price:
                    self.position.close()
                elif self.data.Close[-1] > self.tp_price:
                    self.position.close()
            elif not self.position.is_long:
                if self.data.Close[-1] > self.sl_price:
                    self.position.close()
                elif self.data.Close[-1] < self.tp_price:
                    self.position.close()
