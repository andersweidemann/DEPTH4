import numpy as np
import pandas as pd
from agents.backtester import RegimeStrategy
from agents.signals import sma, ema, atr, rsi, bollinger, bb_width, donchian, atr_breakout_levels, session_mask
from agents.regime import adx, atr_percentile, classify, REGIMES
from agents.risk import lots_by_risk_pct, daily_kill_ok, spread_ok, DailyKillState

class Strategy(RegimeStrategy):
    def init(self):
        self.spec = dict(self._spec)
        self._kill_state = DailyKillState(start_of_day_equity=self._equity_start)
        self.upper_bb, self.lower_bb = self.I(bollinger, self.data.Close, 20, 2)
        self.rsi = self.I(rsi, self.data.Close, 14)
        self.bb_width = self.I(bb_width, self.data.Close, 20)
        self.atr = self.I(atr, self.data.High, self.data.Low, self.data.Close, 14)

    def _regime_ok(self):
        rf = self.spec.get("regime_filter")
        if not rf:
            return True
        if rf["type"] == "bb_width_percentile":
            bb_width_percentile = rf["params"]["percentile"]
            bb_width_now = float(self.bb_width[-1])
            bb_widths = self.bb_width[:len(self.bb_width) - 1]
            threshold = np.percentile(bb_widths, bb_width_percentile)
            return bb_width_now < threshold
        return True

    def _enter_if_signal(self):
        entry_rules = self.spec.get("entry_rules")
        if entry_rules:
            long_condition = entry_rules["long"]["condition"]
            short_condition = entry_rules["short"]["condition"]
            if long_condition == "close < lower_bb && rsi < 10":
                if self.data.Close[-1] < self.lower_bb[-1] and self.rsi[-1] < 10:
                    self.position.enter_long(lots_by_risk_pct(self._spec, self.data, self._equity_start))
                    self.sl_price = self.data.Close[-1] - self.atr[-1] * 1.5
                    self.tp_price = self.upper_bb[-1]
            elif short_condition == "close > upper_bb && rsi > 90":
                if self.data.Close[-1] > self.upper_bb[-1] and self.rsi[-1] > 90:
                    self.position.enter_short(lots_by_risk_pct(self._spec, self.data, self._equity_start))
                    self.sl_price = self.data.Close[-1] + self.atr[-1] * 1.5
                    self.tp_price = self.lower_bb[-1]

    def _manage_open(self):
        exit_cfg = self.spec.get("exit_rules")
        if exit_cfg:
            time_stop = exit_cfg.get("time_stop")
            if time_stop and time_stop["type"] == "bars":
                num_bars = time_stop["params"]["num_bars"]
                if self.position:
                    bars_open = len(self.data) - self.position.entry_bar
                    if bars_open >= num_bars:
                        self.position.close()
            if exit_cfg.get("sl") and exit_cfg["sl"]["type"] == "atr":
                atr_multiplier = exit_cfg["sl"]["params"]["multiplier"]
                if self.position:
                    self.sl_price = self.data.Close[-1] - self.atr[-1] * atr_multiplier if self.position.is_long else self.data.Close[-1] + self.atr[-1] * atr_multiplier
            if exit_cfg.get("tp") and exit_cfg["tp"]["type"] == "opposite_bb":
                if self.position:
                    self.tp_price = self.upper_bb[-1] if self.position.is_long else self.lower_bb[-1]
